311 void BaseCouplingScheme::firstExchange()
312 {
313  PRECICE_TRACE(_timeWindows, getTime());
314  checkCompletenessRequiredActions();
315  PRECICE_ASSERT(_isInitialized, "Before calling advance() coupling scheme has to be initialized via initialize().");
316  _hasDataBeenReceived = false;
317  _isTimeWindowComplete = false;
318 
319  PRECICE_ASSERT(_couplingMode != Undefined);
320 
321  if (reachedEndOfTimeWindow()) {
322 
323  _timeWindows += 1; // increment window counter. If not converged, will be decremented again later.
324 
325  //If preCICE has stopped before the end of the time window we have to duplicate the last available sample and put it at the end of the time window.
326  // We have to exclude the case where coupling scheme does not have a time window size, since this will cause problem with the interpolation later on
327  if (getNextTimeStepMaxSize() > math::NUMERICAL_ZERO_DIFFERENCE && hasTimeWindowSize()) {
328 
329  addTimeStepAtWindowEnd();
330 
331  // Update the _computedTimeWindowPart in order to keep the time within preCICE synchronised
332  // Has to be done before the second exchange, since the serial coupling scheme moves to the new time window before updating _timeWindowStartTime
333  _computedTimeWindowPart = _timeWindowSize;
334  }
335 
336  exchangeFirstData();
337  }
338 }

345 void BaseCouplingScheme::secondExchange()
346 {
347  PRECICE_TRACE(_timeWindows, getTime());
348  checkCompletenessRequiredActions();
349  PRECICE_ASSERT(_isInitialized, "Before calling advance() coupling scheme has to be initialized via initialize().");
350  PRECICE_ASSERT(_couplingMode != Undefined);
351 
352  // from first phase
353  PRECICE_ASSERT(!_isTimeWindowComplete);
354 
355  if (reachedEndOfTimeWindow()) {
356 
357  exchangeSecondData();
358 
359  if (isImplicitCouplingScheme()) { // check convergence
360  if (not hasConverged()) { // repeat window
361  PRECICE_DEBUG("No convergence achieved");
362  requireAction(CouplingScheme::Action::ReadCheckpoint);
363  // The computed time window part equals the time window size, since the
364  // time window remainder is zero. Subtract the time window size and do another
365  // coupling iteration.
366  PRECICE_ASSERT(math::greater(_computedTimeWindowPart, 0.0));
367  _timeWindows -= 1;
368  _isTimeWindowComplete = false;
369  } else { // write output, prepare for next window
370  PRECICE_DEBUG("Convergence achieved");
371  advanceTXTWriters();
372  PRECICE_INFO("Time window completed");
373  _isTimeWindowComplete = true;
374  if (isCouplingOngoing()) {
375  PRECICE_DEBUG("Setting require create checkpoint");
376  requireAction(CouplingScheme::Action::WriteCheckpoint);
377  }
378  }
379  //update iterations
380  _totalIterations++;
381  if (not hasConverged()) {
382  _iterations++;
383  } else {
384  _iterations = 1;
385  }
386  } else {
387  PRECICE_ASSERT(isExplicitCouplingScheme());
388  PRECICE_INFO("Time window completed");
389  _isTimeWindowComplete = true;
390  }
391  if (isCouplingOngoing()) {
392  PRECICE_ASSERT(_hasDataBeenReceived);
393  }
394 
395  // Update internal time tracking
396  if (_isTimeWindowComplete) {
397  _timeWindowStartTime += _timeWindowSize;
398  }
399  _computedTimeWindowPart = 0.0; // reset window
400  _timeWindowSize = _nextTimeWindowSize;
401  }
402 }

659  PRECICE_ASSERT(_allData.count(dataID) == 1, "Data with given data ID must exist!");
660  convMeasure.couplingData = _allData.at(dataID);
661  convMeasure.suffices = suffices;
662  convMeasure.strict = strict;
663  convMeasure.measure = std::move(measure);
664  convMeasure.doesLogging = doesLogging;
665  _convergenceMeasures.push_back(convMeasure);
666 }

668 bool BaseCouplingScheme::measureConvergence()
669 {
670  PRECICE_TRACE();
671  PRECICE_ASSERT(not doesFirstStep());
672  if (not utils::IntraComm::isSecondary()) {
673  _convergenceWriter->writeData("TimeWindow", _timeWindows - 1);
674  _convergenceWriter->writeData("Iteration", _iterations);
675  }
676 
677  // If no convergence measures are defined, we never converge
678  if (_convergenceMeasures.empty()) {
679  PRECICE_INFO("No converge measures defined.");
680  return false;
681  }
682 
683  // There are convergence measures defined, so we need to check them
684  bool allConverged = true;
685  bool oneSuffices = false; // at least one convergence measure suffices and did converge
686  bool oneStrict = false; // at least one convergence measure is strict and did not converge
687 
688  const bool reachedMinIterations = _iterations >= _minIterations;
689  for (const auto &convMeasure : _convergenceMeasures) {
690  PRECICE_ASSERT(convMeasure.couplingData != nullptr);
691  PRECICE_ASSERT(convMeasure.measure.get() != nullptr);
692  PRECICE_ASSERT(convMeasure.couplingData->previousIteration().size() == convMeasure.couplingData->values().size(), convMeasure.couplingData->previousIteration().size(), convMeasure.couplingData->values().size(), convMeasure.couplingData->getDataName());
693  convMeasure.measure->measure(convMeasure.couplingData->previousIteration(), convMeasure.couplingData->values());
694 
695  if (not utils::IntraComm::isSecondary() && convMeasure.doesLogging) {
696  _convergenceWriter->writeData(convMeasure.logHeader(), convMeasure.measure->getNormResidual());
697  }
698 
699  if (not convMeasure.measure->isConvergence()) {
700  allConverged = false;
701  if (convMeasure.strict) {
702  PRECICE_ASSERT(_maxIterations > 0);
703  oneStrict = true;
704  PRECICE_CHECK(_iterations < _maxIterations,
705  "The strict convergence measure for data \"" + convMeasure.couplingData->getDataName() +
706  "\" did not converge within the maximum allowed iterations, which terminates the simulation. "
707  "To avoid this forced termination do not mark the convergence measure as strict.")
708  }
709  } else if (convMeasure.suffices == true) {
710  oneSuffices = true;
711  }
712 
713  PRECICE_INFO(convMeasure.measure->printState(convMeasure.couplingData->getDataName()));
714  }
715 
716  std::string messageSuffix;
717  if (not reachedMinIterations) {
718  messageSuffix = " but hasn't yet reached minimal amount of iterations";
719  }
720  if (allConverged) {
721  PRECICE_INFO("All converged{}", messageSuffix);
722  } else if (oneSuffices && not oneStrict) { // strict overrules suffices
723  PRECICE_INFO("Sufficient measures converged{}", messageSuffix);
724  }
725 
726  return reachedMinIterations && (allConverged || (oneSuffices && not oneStrict));
727 }

821 void BaseCouplingScheme::doImplicitStep()
822 {
823  PRECICE_DEBUG("measure convergence of the coupling iteration");
824  _hasConverged = measureConvergence();
825  // Stop, when maximal iteration count (given in config) is reached
826  if (_iterations == _maxIterations)
827  _hasConverged = true;
828 
829  // coupling iteration converged for current time window. Advance in time.
830  if (_hasConverged) {
831  if (_acceleration) {
832  _acceleration->iterationsConverged(getAccelerationData());
833  }
834  newConvergenceMeasurements();
835  } else {
836  // no convergence achieved for the coupling iteration within the current time window
837  if (_acceleration) {
838  // Acceleration works on CouplingData::values(), so we retrieve the data from the storage, perform the acceleration and then put the data back into the storage. See also https://github.com/precice/precice/issues/1645.
839  // @todo For acceleration schemes as described in "Rüth, B, Uekermann, B, Mehl, M, Birken, P, Monge, A, Bungartz, H-J. Quasi-Newton waveform iteration for partitioned surface-coupled multiphysics applications. https://doi.org/10.1002/nme.6443" we need a more elaborate implementation.
840 
841  // Load from storage into buffer
842  for (auto &data : getAccelerationData() | boost::adaptors::map_values) {
843  const auto &stamples = data->stamples();
844  PRECICE_ASSERT(stamples.size() > 0);
845  data->sample() = stamples.back().sample;
846  }
847 
848  _acceleration->performAcceleration(getAccelerationData());
849 
850  // Store from buffer
851  // @todo Currently only data at end of window is accelerated. Remaining data in storage stays as it is.
852  for (auto &data : getAccelerationData() | boost::adaptors::map_values) {
853  data->setSampleAtTime(getTime(), data->sample());
854  }
855  }
856  }
857 }
