711 std::string KSPSolver::summaryFor(Vector &b)
712 {
713  // See PETSc manual page for KSPGetConvergedReason to understand this function
714  // We treat divergence due to reaching max iterations as "stopped"
715  KSPConvergedReason convReason;
716  KSPGetConvergedReason(ksp, &convReason);
717 
718  PetscReal rtol, atol, dtol;
719  PetscInt miter;
720  KSPGetTolerances(ksp, &rtol, &atol, &dtol, &miter);
721 
722  std::ostringstream oss;
723  {
724  bool converged = (convReason >= 0);
725  bool stopped = (convReason == KSP_DIVERGED_ITS);
726  oss << "Solver " << (converged ? "converged" : (stopped ? "stopped" : "diverged"));
727  }
728  oss << " after " << getIterationNumber() << " of " << miter << " iterations due to";
729 
730  switch (convReason) {
731  case (KSP_CONVERGED_RTOL):
732  case (KSP_CONVERGED_RTOL_NORMAL):
733  oss << " sufficient relative convergence";
