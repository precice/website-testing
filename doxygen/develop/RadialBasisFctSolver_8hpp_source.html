82  const std::array<bool, 3> & activeAxis = {{true, true, true}})

93 template <typename IndexContainer>
94 constexpr void reduceActiveAxis(const mesh::Mesh &mesh, const IndexContainer &IDs, std::array<bool, 3> &axis)
95 {
96  // make a pair of the axis and the difference
97  std::array<std::pair<int, double>, 3> differences;
98 
99  // Compute the difference magnitude per direction
100  for (std::size_t d = 0; d < axis.size(); ++d) {
101  // Ignore dead axis here, i.e., apply the max value such that they are sorted on the last position(s)
102  if (axis[d] == false) {
103  differences[d] = std::make_pair<int, double>(d, std::numeric_limits<double>::max());
104  } else {
105  auto res = std::minmax_element(IDs.begin(), IDs.end(), [&](const auto &a, const auto &b) { return mesh.vertices()[a].rawCoords()[d] < mesh.vertices()[b].rawCoords()[d]; });
106  // Check if we are above or below the threshold
107  differences[d] = std::make_pair<int, double>(d, std::abs(mesh.vertices()[*res.second].rawCoords()[d] - mesh.vertices()[*res.first].rawCoords()[d]));
108  }
109  }
110 
111  std::sort(differences.begin(), differences.end(), [](const auto &d1, const auto &d2) { return d1.second < d2.second; });
112  // Disable the axis having the smallest expansion
113  axis[differences[0].first] = false;
114 }

118 inline void fillPolynomialEntries(Eigen::MatrixXd &matrix, const mesh::Mesh &mesh, const IndexContainer &IDs, Eigen::Index startIndex, std::array<bool, 3> activeAxis)
119 {
120  // Loop over all vertices in the mesh
121  for (const auto &i : IDs | boost::adaptors::indexed()) {
122 
123  // 1. the constant contribution
124  matrix(i.index(), startIndex) = 1.0;
125 
126  // 2. the linear contribution
127  const auto & u = mesh.vertices()[i.value()].rawCoords();
128  unsigned int k = 0;
129  // Loop over all three space dimension and ignore dead axis
130  for (unsigned int d = 0; d < activeAxis.size(); ++d) {
131  if (activeAxis[d]) {
132  PRECICE_ASSERT(matrix.rows() > i.index(), matrix.rows(), i.index());
133  PRECICE_ASSERT(matrix.cols() > startIndex + 1 + k, matrix.cols(), startIndex + 1 + k);
134  matrix(i.index(), startIndex + 1 + k) = u[d];
135  ++k;
136  }
137  }
138  }

142 Eigen::MatrixXd buildMatrixCLU(RADIAL_BASIS_FUNCTION_T basisFunction, const mesh::Mesh &inputMesh, const IndexContainer &inputIDs,
143  std::array<bool, 3> activeAxis, Polynomial polynomial)
144 {
145  // Treat the 2D case as 3D case with dead axis
146  const unsigned int deadDimensions = std::count(activeAxis.begin(), activeAxis.end(), false);
147  const unsigned int dimensions = 3;
148  const unsigned int polyparams = polynomial == Polynomial::ON ? 1 + dimensions - deadDimensions : 0;
149 
150  // Add linear polynom degrees if polynomial requires this
151  const auto inputSize = inputIDs.size();
152  const auto n = inputSize + polyparams;
153 
154  PRECICE_ASSERT((inputMesh.getDimensions() == 3) || activeAxis[2] == false);
155  PRECICE_ASSERT((inputSize >= 1 + polyparams) || polynomial != Polynomial::ON, inputSize);
156 
157  precice::profiling::Event cluAssembly{"map.rbf.assembleSystemMatrix"};
158  Eigen::MatrixXd matrixCLU(n, n);
159 
160  // Required to fill the poly -> poly entries in the matrix, which remain otherwise untouched
161  if (polynomial == Polynomial::ON) {
162  matrixCLU.setZero();
163  }
164 
165  // Compute RBF matrix entries
166  auto i_iter = inputIDs.begin();
167  Eigen::Index i_index = 0;
168  for (; i_iter != inputIDs.end(); ++i_iter, ++i_index) {
169  const auto &u = inputMesh.vertices()[*i_iter].rawCoords();
170  auto j_iter = i_iter;
171  auto j_index = i_index;
172  for (; j_iter != inputIDs.end(); ++j_iter, ++j_index) {
173  const auto &v = inputMesh.vertices()[*j_iter].rawCoords();
174  double squaredDifference = computeSquaredDifference(u, v, activeAxis);
175  matrixCLU(i_index, j_index) = basisFunction.evaluate(std::sqrt(squaredDifference));
176  }
177  }
178  cluAssembly.stop();
179 
180  // Add potentially the polynomial contribution in the matrix
181  if (polynomial == Polynomial::ON) {
182  fillPolynomialEntries(matrixCLU, inputMesh, inputIDs, inputSize, activeAxis);
183  }
184  matrixCLU.triangularView<Eigen::Lower>() = matrixCLU.transpose();
185  return matrixCLU;

227 template <typename IndexContainer>
228 RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>::RadialBasisFctSolver(RADIAL_BASIS_FUNCTION_T basisFunction, const mesh::Mesh &inputMesh, const IndexContainer &inputIDs,
229  const mesh::Mesh &outputMesh, const IndexContainer &outputIDs, std::vector<bool> deadAxis, Polynomial polynomial)
230 {
231  PRECICE_ASSERT(!(RADIAL_BASIS_FUNCTION_T::isStrictlyPositiveDefinite() && polynomial == Polynomial::ON), "The integrated polynomial (polynomial=\"on\") is not supported for the selected radial-basis function. Please select another radial-basis function or change the polynomial configuration.");
232  // Convert dead axis vector into an active axis array so that we can handle the reduction more easily
233  std::array<bool, 3> activeAxis({{false, false, false}});
234  std::transform(deadAxis.begin(), deadAxis.end(), activeAxis.begin(), [](const auto ax) { return !ax; });
235 
236  // First, assemble the interpolation matrix and check the invertability
237  bool decompositionSuccessful = false;
238  if constexpr (RADIAL_BASIS_FUNCTION_T::isStrictlyPositiveDefinite()) {
239  _decMatrixC = buildMatrixCLU(basisFunction, inputMesh, inputIDs, activeAxis, polynomial).llt();
240  decompositionSuccessful = _decMatrixC.info() == Eigen::ComputationInfo::Success;
241  } else {
242  _decMatrixC = buildMatrixCLU(basisFunction, inputMesh, inputIDs, activeAxis, polynomial).colPivHouseholderQr();
243  decompositionSuccessful = _decMatrixC.isInvertible();
244  }
245 
246  PRECICE_CHECK(decompositionSuccessful,
247  "The interpolation matrix of the RBF mapping from mesh \"{}\" to mesh \"{}\" is not invertable. "
248  "This means that the mapping problem is not well-posed. "
249  "Please check if your coupling meshes are correct (e.g. no vertices are duplicated) or reconfigure "
250  "your basis-function (e.g. reduce the support-radius).",
251  inputMesh.getName(), outputMesh.getName());
252 
253  // Second, assemble evaluation matrix
254  _matrixA = buildMatrixA(basisFunction, inputMesh, inputIDs, outputMesh, outputIDs, activeAxis, polynomial);
255 
256  // In case we deal with separated polynomials, we need dedicated matrices for the polynomial contribution
257  if (polynomial == Polynomial::SEPARATE) {
258 
259  // 4 = 1 + dimensions(3) = maximum number of polynomial parameters
260  auto localActiveAxis = activeAxis;
261  unsigned int polyParams = 4 - std::count(localActiveAxis.begin(), localActiveAxis.end(), false);
262 
263  do {
264  // First, build matrix Q and check for the condition number
265  _matrixQ.resize(inputIDs.size(), polyParams);
266  fillPolynomialEntries(_matrixQ, inputMesh, inputIDs, 0, localActiveAxis);
267 
268  // Compute the condition number
269  Eigen::JacobiSVD<Eigen::MatrixXd> svd(_matrixQ);
270  PRECICE_ASSERT(svd.singularValues().size() > 0);
271  PRECICE_DEBUG("Singular values in polynomial solver: {}", svd.singularValues());
272  const double conditionNumber = svd.singularValues()(0) / std::max(svd.singularValues()(svd.singularValues().size() - 1), math::NUMERICAL_ZERO_DIFFERENCE);
273  PRECICE_DEBUG("Condition number: {}", conditionNumber);
274 
275  // Disable one axis
276  if (conditionNumber > 1e5) {
277  reduceActiveAxis(inputMesh, inputIDs, localActiveAxis);
278  polyParams = 4 - std::count(localActiveAxis.begin(), localActiveAxis.end(), false);
279  } else {
280  break;
281  }
282  } while (true);
283 
284  // allocate and fill matrix V for the outputMesh
285  _matrixV.resize(outputIDs.size(), polyParams);
286  fillPolynomialEntries(_matrixV, outputMesh, outputIDs, 0, localActiveAxis);
287 
288  // 3. compute decomposition
289  _qrMatrixQ = _matrixQ.colPivHouseholderQr();
290  }

293 template <typename RADIAL_BASIS_FUNCTION_T>
294 Eigen::VectorXd RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>::solveConservative(const Eigen::VectorXd &inputData, Polynomial polynomial) const
295 {
296  PRECICE_ASSERT((_matrixV.size() > 0 && polynomial == Polynomial::SEPARATE) || _matrixV.size() == 0, _matrixV.size());
297  // TODO: Avoid temporary allocations
298  // Au is equal to the eta in our PETSc implementation
299  PRECICE_ASSERT(inputData.size() == _matrixA.rows());
300  Eigen::VectorXd Au = _matrixA.transpose() * inputData;
301  PRECICE_ASSERT(Au.size() == _matrixA.cols());
302 
303  // mu in the PETSc implementation
304  Eigen::VectorXd out = _decMatrixC.solve(Au);
305 
306  if (polynomial == Polynomial::SEPARATE) {
307  Eigen::VectorXd epsilon = _matrixV.transpose() * inputData;
308  PRECICE_ASSERT(epsilon.size() == _matrixV.cols());
309 
310  // epsilon = Q^T * mu - epsilon (tau in the PETSc impl)
311  epsilon -= _matrixQ.transpose() * out;
312  PRECICE_ASSERT(epsilon.size() == _matrixQ.cols());
313 
314  // out = out - solveTranspose tau (sigma in the PETSc impl)
315  // Newer version of eigen provide the solve() for transpose() matrix decopmositions
316 #if EIGEN_VERSION_AT_LEAST(3, 4, 0)
317  out -= static_cast<Eigen::VectorXd>(_qrMatrixQ.transpose().solve(-epsilon));
318 #else
319  // Backwards compatible version
320  Eigen::VectorXd sigma(_matrixQ.rows());
321  const Eigen::Index nonzero_pivots = _qrMatrixQ.nonzeroPivots();
322 
323  if (nonzero_pivots == 0) {
324  sigma.setZero();
325  } else {
326  Eigen::VectorXd c(_qrMatrixQ.colsPermutation().transpose() * (-epsilon));
327 
328  _qrMatrixQ.matrixQR().topLeftCorner(nonzero_pivots, nonzero_pivots).template triangularView<Eigen::Upper>().transpose().conjugate().solveInPlace(c.topRows(nonzero_pivots));
329 
330  sigma.topRows(nonzero_pivots) = c.topRows(nonzero_pivots);
331  sigma.bottomRows(_qrMatrixQ.rows() - nonzero_pivots).setZero();
332 
333  sigma.applyOnTheLeft(_qrMatrixQ.householderQ().setLength(nonzero_pivots));
334  out -= sigma;
335  }
336 #endif
337  }
338  return out;
339 }

341 template <typename RADIAL_BASIS_FUNCTION_T>
342 Eigen::VectorXd RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>::solveConsistent(Eigen::VectorXd &inputData, Polynomial polynomial) const
343 {
344  PRECICE_ASSERT((_matrixQ.size() > 0 && polynomial == Polynomial::SEPARATE) || _matrixQ.size() == 0);
345  Eigen::VectorXd polynomialContribution;
346  // Solve polynomial QR and subtract it from the input data
347  if (polynomial == Polynomial::SEPARATE) {
348  polynomialContribution = _qrMatrixQ.solve(inputData);
349  inputData -= (_matrixQ * polynomialContribution);
350  }
351 
352  // Integrated polynomial (and separated)
353  PRECICE_ASSERT(inputData.size() == _matrixA.cols());
354  Eigen::VectorXd p = _decMatrixC.solve(inputData);
355  PRECICE_ASSERT(p.size() == _matrixA.cols());
356  Eigen::VectorXd out = _matrixA * p;
357 
358  // Add the polynomial part again for separated polynomial
359  if (polynomial == Polynomial::SEPARATE) {
360  out += (_matrixV * polynomialContribution);
361  }
362  return out;
363 }
