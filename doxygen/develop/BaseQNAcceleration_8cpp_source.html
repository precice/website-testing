290

300

304  _oldResiduals = _residuals; // Store current residual
305 
306  // Perform constant relaxation
307  // with residual: x_new = x_old + omega * res
308  _residuals *= _initialRelaxation;
309  _residuals += _oldValues;
310  _values = _residuals;
311 
312  computeUnderrelaxationSecondaryData(cplData);
313  } else {
314  PRECICE_DEBUG(" Performing quasi-Newton Step");
315 
316  // If the previous time window converged within one single iteration, nothing was added
317  // to the LS system matrices and they need to be restored from the backup at time T-2
318  if (not _firstTimeWindow && (getLSSystemCols() < 1) && (_timeWindowsReused == 0) && not _forceInitialRelaxation) {
319  PRECICE_DEBUG(" Last time window converged after one iteration. Need to restore the matrices from backup.");
320

322  _matrixV = _matrixVBackup;
323  _matrixW = _matrixWBackup;
324 
325  // re-computation of QR decomposition from _matrixV = _matrixVBackup
326  // this occurs very rarely, to be precise, it occurs only if the coupling terminates
327  // after the first iteration and the matrix data from time window t-2 has to be used
328  _preconditioner->apply(_matrixV);
329  _qrV.reset(_matrixV, getLSSystemRows());

331  _resetLS = true; // need to recompute _Wtil, Q, R (only for IMVJ efficient update)
332  }
333

354  _nbDropCols = 0;

356

359  applyFilter();

361 
362  // revert scaling of V, in computeQNUpdate all data objects are unscaled.
363  _preconditioner->revert(_matrixV);
364

371  computeQNUpdate(cplData, xUpdate);
372 
376  _values = _oldValues + xUpdate + _residuals; // = x^k + delta_x + r^k - q^k
377 
378  // pending deletion: delete old V, W matrices if timeWindowsReused = 0
379  // those were only needed for the first iteration (instead of underrelax.)
380  if (_firstIteration && _timeWindowsReused == 0 && not _forceInitialRelaxation) {
381  // save current matrix data in case the coupling for the next time window will terminate
382  // after the first iteration (no new data, i.e., V = W = 0)
383  if (getLSSystemCols() > 0) {
384  _matrixColsBackup = _matrixCols;
385  _matrixVBackup = _matrixV;
386  _matrixWBackup = _matrixW;
387  }
388  // if no time windows reused, the matrix data needs to be cleared as it was only needed for the
389  // QN-step in the first iteration (idea: rather perform QN-step with information from last converged
390  // time window instead of doing a underrelaxation)

393  _matrixW.resize(0, 0);
394  _matrixCols.clear();
395  _matrixCols.push_front(0); // vital after clear()
396  _qrV.reset();
397  // set the number of global rows in the QRFactorization.
398  _qrV.setGlobalRows(getLSSystemRows());

400  }
401  }
402

404  PRECICE_ERROR("The quasi-Newton update contains NaN values. This means that the quasi-Newton acceleration failed to converge. "
405  "When writing your own adapter this could indicate that you give wrong information to preCICE, such as identical "
406  "data in succeeding iterations. Or you do not properly save and reload checkpoints. "
407  "If you give the correct data this could also mean that the coupled problem is too hard to solve. Try to use a QR "
408  "filter or increase its threshold (larger epsilon).");
409  }
410  }
411 
412  splitCouplingData(cplData);
413  // number of iterations (usually equals number of columns in LS-system)
414  its++;
415  _firstIteration = false;
