165  eClusters.stop();

171  // Here, the VertexCluster computes the matrix decompositions directly in case the cluster is non-empty
172  mesh::Mesh centerMesh("pou-centers-" + inMesh->getName(), this->getDimensions(), mesh::Mesh::MESH_ID_UNDEFINED);

180  // of the cluster within the _clusters vector. That's required for the indexing further down and asserted below

188  meshVertices.emplace_back(std::move(center));
189  _clusters.emplace_back(std::move(cluster));
190  }
191  }

194  // Log the average number of resulting clusters
195  PRECICE_DEBUG("Partition of unity data mapping between mesh \"{}\" and mesh \"{}\": mesh \"{}\" on rank {} was decomposed into {} clusters.", this->input()->getName(), this->output()->getName(), inMesh->getName(), utils::IntraComm::getRank(), _clusters.size());
196 
197  if (_clusters.size() > 0) {

207 
208  // Step 3: index the clusters / the center mesh in order to define the output vertex -> cluster ownership
209  // the ownership is required to compute the normalized partition of unity weights (Step 4)
210  query::Index clusterIndex(centerMesh);
211  // Step 4: find all clusters the output vertex lies in, i.e., find all cluster centers which have the distance of a cluster radius from the given output vertex
212  // Here, we do this using the RTree on the centerMesh: VertexID (queried from the centersMesh) == clusterID, by construction above. The loop uses
213  // the vertices to compute the weights required for the partition of unity data mapping.
214  // Note: this could also be done on-the-fly in the map data phase for dynamic queries, which would require to make the mesh as well as the indexTree member variables.
215  PRECICE_DEBUG("Computing cluster-vertex association");
216  for (const auto &vertex : outMesh->vertices()) {
217  // Step 4a: get the relevant clusters for the output vertex
218  auto clusterIDs = clusterIndex.getVerticesInsideBox(vertex, _clusterRadius);
219  const auto localNumberOfClusters = clusterIDs.size();
220  // Consider the case where we didn't find any cluster (meshes don't match very well)
221  if (localNumberOfClusters == 0) {
222  PRECICE_ERROR("Output vertex {} of mesh \"{}\" could not be assigned to any cluster in the rbf-pum mapping. This probably means that the meshes do not match well geometry-wise: Visualize the exported preCICE meshes to confirm."
223  " If the meshes are fine geometry-wise, you can try to increase the number of \"vertices-per-cluster\" (default is 50), the \"relative-overlap\" (default is 0.15),"
224  " or disable the option \"project-to-input\"."
225  "These options are only valid for the <mapping:rbf-pum-direct/> tag.",
226  vertex.getCoords(), outMesh->getName());
227  // In principle, we could assign the vertex to the closest cluster using clusterIDs.emplace_back(clusterIndex.getClosestVertex(vertex.getCoords()).index);
228  // However, this leads to a conflict with weights already set in the corresponding cluster, since we insert the ID and, later on, map the ID to a local weight index
229  // Of course, we could rearrange the weights, but we want to avoid the case here anyway, i.e., prefer to abort.
230  }
231 
232  // Next we compute the normalized weights of each output vertex for each partition
233  PRECICE_ASSERT(localNumberOfClusters > 0, "No cluster found for vertex {}", vertex.getCoords());
234 
235  // Step 4b: compute the weight in each partition individually and store them in 'weights'
236  std::vector<double> weights(localNumberOfClusters);
237  std::transform(clusterIDs.cbegin(), clusterIDs.cend(), weights.begin(), [&](const auto &ids) { return _clusters[ids].computeWeight(vertex); });
238  double weightSum = std::accumulate(weights.begin(), weights.end(), static_cast<double>(0.));
239  // TODO: This covers the edge case of vertices being at the edge of (several) clusters
240  // In case the sum is equal to zero, we assign equal weights for all clusters
241  if (weightSum <= 0) {
242  PRECICE_ASSERT(weights.size() > 0);
243  std::for_each(weights.begin(), weights.end(), [&weights](auto &w) { w = 1. / weights.size(); });
244  weightSum = 1;
245  }
246  PRECICE_ASSERT(weightSum > 0);
247 
248  // Step 4c: scale the weight using the weight sum and store the normalized weight in all associated clusters
249  for (unsigned int i = 0; i < localNumberOfClusters; ++i) {
250  PRECICE_ASSERT(clusterIDs[i] < static_cast<int>(_clusters.size()));
251  _clusters[clusterIDs[i]].setNormalizedWeight(weights[i] / weightSum, vertex.getID());
252  }
253  }
254  eWeights.stop();

257  // exportClusterCentersAsVTU(centerMesh);
258

293 void PartitionOfUnityMapping<RADIAL_BASIS_FUNCTION_T>::tagMeshFirstRound()
294 {
295  PRECICE_TRACE();
296  mesh::PtrMesh filterMesh, outMesh;
297  if (this->hasConstraint(Mapping::CONSERVATIVE)) {
298  filterMesh = this->output(); // remote
299  outMesh = this->input(); // local
300  } else {
301  filterMesh = this->input(); // remote
302  outMesh = this->output(); // local
303  }
304 
305  if (outMesh->vertices().empty())
306  return; // Ranks not at the interface should never hold interface vertices
307 
308  // Note that we don't use the corresponding bounding box functions from
309  // precice::mesh (e.g. ::getBoundingBox), as the stored bounding box might
310  // have the wrong size (e.g. direct access)
311  precice::mesh::BoundingBox bb = filterMesh->index().getRtreeBounds();
312 
313 #ifndef NDEBUG
314  // Safety check
315  precice::mesh::BoundingBox bb_check(filterMesh->getDimensions());
316  for (const mesh::Vertex &vertex : filterMesh->vertices()) {
317  bb_check.expandBy(vertex);
318  }
319  PRECICE_ASSERT(bb_check == bb);
320 #endif
321  // @TODO: This is assert and the function might run into problems if we have only a single.
322  // vertex in the received mesh
323  // However, with the current BB implementation, the expandBy function will just do nothing.
324  PRECICE_ASSERT(!bb.empty());
325  // This function behaves differently when disabling the geometric filtering
326  // without filter, we look at the complete mesh, whereas otherwise, we look at
327  // a fraction of the mesh and we might end up with too few vertices per rank
328  // We cannot prevent too few vertices from being tagged, if we have filtered too much
329  // vertices, but the user could increase the safety-factor or disable the filtering.
330  // When no geometric filter is applid, vertices().size() is here the same as
331  // getGlobalNumberOfVertices
332  if (filterMesh->vertices().size() < _verticesPerCluster &&
333  filterMesh->vertices().size() < static_cast<std::size_t>(filterMesh->getGlobalNumberOfVertices())) {
334  PRECICE_WARN("The repartitioning of the received mesh \"{}\" resulted in {} vertices on this "
335  "rank, which is less than the desired number of vertices per cluster configured "
336  "in the partition of unity mapping ({}). Consider increasing the safety-factor "
337  "or switching off the geometric filter (<receive-mesh: ... geometric-filter=\"no-filter\" .../>)",
338  filterMesh->getName(), filterMesh->vertices().size(), _verticesPerCluster);
339  }
340 
341  if (_clusterRadius == 0)
342  _clusterRadius = impl::estimateClusterRadius(_verticesPerCluster, filterMesh, bb);
343 
344  PRECICE_DEBUG("Cluster radius estimate: {}", _clusterRadius);
345  PRECICE_ASSERT(_clusterRadius > 0);
346 
347  // Get the local bounding boxes
348  auto localBB = outMesh->getBoundingBox();
349  // Now we extend the bounding box by the radius
350  localBB.expandBy(2 * _clusterRadius);
351 
352  // ... and tag all affected vertices
353  auto verticesNew = filterMesh->index().getVerticesInsideBox(localBB);
354 
355  std::for_each(verticesNew.begin(), verticesNew.end(), [&filterMesh](VertexID v) { filterMesh->vertices()[v].tag(); });
356 }
