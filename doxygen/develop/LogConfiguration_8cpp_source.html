183 {
184  if (getGlobalLoggingConfig().locked)
185  return;
186 
187  namespace bl = boost::log;
188  bl::register_formatter_factory("TimeStamp", boost::make_shared<timestamp_formatter_factory>());
189  bl::register_formatter_factory("ColorizedSeverity", boost::make_shared<colorized_severity_formatter_factory>());
190  bl::register_formatter_factory("Severity", boost::make_shared<severity_formatter_factory>());
191  bl::register_simple_filter_factory<bl::trivial::severity_level, char>("Severity");
192 
193  // Possible, longer output format. Currently unused.
194  auto fmtStream =
195  bl::expressions::stream
196  << "(" << bl::expressions::attr<int>("Rank") << ") "
197  << bl::expressions::format_date_time<boost::posix_time::ptime>("TimeStamp", "%H:%M:%S") << " "
198  << bl::expressions::attr<std::string>("File") << ":"
199  << bl::expressions::attr<int>("Line")
200  << " [" << bl::expressions::attr<std::string>("Module") << "] in "
201  << bl::expressions::attr<std::string>("Function") << ": "
202  << bl::expressions::message;
203 
204  // Remove active preCICE sinks
205  using sink_ptr = typename boost::shared_ptr<boost::log::sinks::sink>;
206 
207  static std::vector<sink_ptr> activeSinks;
208  for (auto &sink : activeSinks) {
209  boost::log::core::get()->remove_sink(sink);
210  sink->flush();
211  sink.reset();
212  }
213  activeSinks.clear();
214 
215  // If logging sinks are disabled, then we need to disable the default sink.
216  // We do this by adding a NullSink.
217  // We need to exit after the sink removal as the default sink exists before
218  // the log configuration is parsed.
219  if (auto noconfigs = std::none_of(configs.begin(), configs.end(), [](const auto &config) { return config.enabled; });
220  !enabled && noconfigs) {
221  auto sink = boost::make_shared<NullSink>();
222  boost::log::core::get()->add_sink(sink);
223  activeSinks.emplace_back(std::move(sink));
224  return;
225  }
226 
227  // Add the default config in case no sinks are configured
228  if (configs.empty()) {
229  configs.emplace_back();
230  }
231 
232  // Create new sinks
233  for (const auto &config : configs) {
234  if (!config.enabled) {
235  continue;
236  }
237 
238  // Setup backend of sink
239  boost::shared_ptr<StreamBackend> backend;
240  if (config.type == "file")
241  backend = boost::make_shared<StreamBackend>(boost::shared_ptr<std::ostream>(new std::ofstream(config.output)));
242  if (config.type == "stream") {
243  if (config.output == "stdout")
244  backend = boost::make_shared<StreamBackend>(boost::shared_ptr<std::ostream>(&std::cout, boost::null_deleter()));
245  if (config.output == "stderr")
246  backend = boost::make_shared<StreamBackend>(boost::shared_ptr<std::ostream>(&std::cerr, boost::null_deleter()));
247  }
248  PRECICE_ASSERT(backend != nullptr, "The logging backend was not initialized properly. Check your log config.");
249  backend->auto_flush(true);
250 
251  // Setup sink
252  auto sink = boost::make_shared<boost::log::sinks::synchronous_sink<StreamBackend>>(backend);
253  sink->set_formatter(boost::log::parse_formatter(config.format));
254 
255  if (config.filter.empty()) {
256  sink->set_filter(boost::log::expressions::attr<bool>("preCICE") == true);
257  } else {
258  // We extend the filter here to filter all log entries not originating from preCICE.
259  sink->set_filter(boost::log::parse_filter("%preCICE% & ( " + config.filter + " )"));
260  }
261 
262  boost::log::core::get()->add_sink(sink);
263  activeSinks.emplace_back(std::move(sink));
264  }
265 }
