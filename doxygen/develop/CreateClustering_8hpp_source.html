99 std::vector<VertexID> getNeighborsWithinSphere(const Vertices &vertices, VertexID centerID, double radius, const ArrayType neighborOffsets)
100 {
101  // number of neighbors = (3^dim)-1
102  static_assert((neighborOffsets.size() == 26) || (neighborOffsets.size() == 8));
103  std::vector<VertexID> result;
104  for (auto off : neighborOffsets) {
105  // corner case where we have 'dead axis' in the bounding box
106  if (off == 0) {
107  continue;
108  }
109  // compute neighbor ID
110  // @todo target type should be an unsigned int
111  PRECICE_ASSERT(off != 0);
112  // Assumption, vertex position in the container == vertex ID
113  VertexID neighborID = centerID + off;
114  // We might run out of the array bounds along the edges, so we only consider vertices inside
115  if (neighborID >= 0 && neighborID < static_cast<int>(vertices.size())) {
116  auto &neighborVertex = vertices[neighborID];
117  auto &center = vertices[centerID];
118  if (!neighborVertex.isTagged() && (computeSquaredDifference(center.rawCoords(), neighborVertex.rawCoords()) < math::pow_int<2>(radius))) {
119  result.emplace_back(neighborID);
120  }
121  }
122  }
123  return result;
124 }

225 inline double estimateClusterRadius(unsigned int verticesPerCluster, mesh::PtrMesh inMesh, const mesh::BoundingBox &bb)
226 {
227  // Step 1: Generate random samples from the input mesh
228  std::vector<VertexID> randomSamples;
229 
230  PRECICE_ASSERT(!bb.empty(), "Center computation won't work right now.");
231  // All samples are 'moved' to the closest vertex from the input mesh in order
232  // to avoid empty clusters when we have shell-like meshes (as in surface coupling)
233  // The first sample: the vertex closest to the bounding box center
234  const auto bbCenter = bb.center();
235  randomSamples.emplace_back(inMesh->index().getClosestVertex(bbCenter).index);
236  // Now we place samples for each space dimension above and below the bounding box center
237  // and look for the closest vertex in the input mesh
238  // In 2D the sampling would like this
239  // +---------------------+
240  // | x |
241  // | |
242  // | x c x |
243  // | |
244  // | x |
245  // +---------------------+
246 
247  for (int d = 0; d < inMesh->getDimensions(); ++d) {
248  auto sample = bbCenter;
249  // Lower as the center
250  sample[d] -= bb.getEdgeLength(d) * 0.25;
251  randomSamples.emplace_back(inMesh->index().getClosestVertex(sample).index);
252  // Higher than the center
253  sample[d] += bb.getEdgeLength(d) * 0.5;
254  randomSamples.emplace_back(inMesh->index().getClosestVertex(sample).index);
255  }
256 
257  // Step 2: Compute the radius of the randomSamples ('centers'), which would have verticesPerCluster vertices
258  std::vector<double> sampledClusterRadii;
259  for (auto s : randomSamples) {
260  // ask the index tree for the k-nearest neighbors in order to estimate the point density
261  auto kNearestVertexIDs = inMesh->index().getClosestVertices(inMesh->vertices()[s].getCoords(), verticesPerCluster);
262  // compute the distance of each point to the center
263  std::vector<double> squaredRadius(kNearestVertexIDs.size());
264  std::transform(kNearestVertexIDs.begin(), kNearestVertexIDs.end(), squaredRadius.begin(), [&inMesh, s](auto i) {
265  return computeSquaredDifference(inMesh->vertices()[i].rawCoords(), inMesh->vertices()[s].rawCoords());
266  });
267  // Store the maximum distance
268  auto maxRadius = std::max_element(squaredRadius.begin(), squaredRadius.end());
269  sampledClusterRadii.emplace_back(std::sqrt(*maxRadius));
270  }
271 
272  // Step 3: Sort (using nth_element) the sampled radii and select the median as cluster radius
273  PRECICE_ASSERT(sampledClusterRadii.size() % 2 != 0, "Median calculation is only valid for odd number of elements.");
274  PRECICE_ASSERT(sampledClusterRadii.size() > 0);
275  unsigned int middle = sampledClusterRadii.size() / 2;
276  std::nth_element(sampledClusterRadii.begin(), sampledClusterRadii.begin() + middle, sampledClusterRadii.end());
277  double clusterRadius = sampledClusterRadii[middle];
278 
279  return clusterRadius;
280 }

304 inline std::tuple<double, Vertices> createClustering(mesh::PtrMesh inMesh, mesh::PtrMesh outMesh,
305  double relativeOverlap, unsigned int verticesPerCluster,
306  bool projectClustersToInput)
307 {
308  precice::logging::Logger _log{"impl::createClustering"};
309  PRECICE_TRACE();
310  PRECICE_ASSERT(relativeOverlap < 1);
311  PRECICE_ASSERT(verticesPerCluster > 0);
312  PRECICE_ASSERT(inMesh->getDimensions() == outMesh->getDimensions());
313 
314  // If we have either no input or no output vertices, we return immediately
315  if (inMesh->vertices().size() == 0 || outMesh->vertices().size() == 0)
316  return {double{}, Vertices{}};
317 
318  PRECICE_ASSERT(!outMesh->vertices().empty() && !inMesh->vertices().empty());
319 
320  // startGridAtEdge boolean switch in order to decide either to start the clustering at the edge of the bounding box in each direction
321  // (true) or start the clustering inside the bounding box (edge + 0.5 radius). The latter approach leads to fewer clusters,
322  // but might result in a worse clustering
323  const bool startGridAtEdge = projectClustersToInput;
324 
325  PRECICE_DEBUG("Relative overlap: {}", relativeOverlap);
326  PRECICE_DEBUG("Vertices per cluster: {}", verticesPerCluster);
327 
328  // Step 1: Compute the local bounding box of the input mesh manually
329  // Note that we don't use the corresponding bounding box functions from
330  // precice::mesh (e.g. ::getBoundingBox), as the stored bounding box might
331  // have the wrong size (e.g. direct access)
332  // @todo: Which mesh should be used in order to determine the cluster centers:
333  // pro outMesh: we want perfectly fitting clusters around our output vertices
334  // however, this makes the cluster distribution/mapping dependent on the output
335  precice::mesh::BoundingBox localBB = inMesh->index().getRtreeBounds();
336 
337 #ifndef NDEBUG
338  // Safety check
339  precice::mesh::BoundingBox bb_check(inMesh->getDimensions());
340  for (const mesh::Vertex &vertex : inMesh->vertices()) {
341  bb_check.expandBy(vertex);
342  }
343  PRECICE_ASSERT(bb_check == localBB);
344 #endif
345 
346  // If we have less vertices in the whole domain than our target cluster size,
347  // we just use a single cluster. The clustering result of the algorithm further
348  // down is in this case not optimal.
349  // The single cluster has in principle a radius of inf. We use here twice the
350  // length of the longest bounding box edge length and the center of the bounding
351  // box for the center point.
352  if (inMesh->vertices().size() < verticesPerCluster)
353  return {localBB.longestEdgeLength() * 2, Vertices{mesh::Vertex({localBB.center(), 0})}};
354 
355  // We define a convenience alias for the localBB. In case we need to synchronize the clustering across ranks later on, we need
356  // to work with the global bounding box of the whole domain.
357  auto globalBB = localBB;
358 
359  // Step 2: Now we pick random samples from the input mesh and ask the index tree for the k-nearest neighbors
360  // in order to estimate the point density and determine a proper cluster radius (see also the function documentation)
361  double clusterRadius = estimateClusterRadius(verticesPerCluster, inMesh, localBB);
362  PRECICE_DEBUG("Vertex cluster radius: {}", clusterRadius);
363 
364  // maximum distance between cluster centers lying diagonal to each other. The maximum distance takes the overlap condition into
365  // account: if the distance between the centers is sqrt(2) * radius, we violate the overlap condition between diagonal clusters
366  // 0.3 should be a good default value
367  const double maximumCenterDistance = std::sqrt(2) * clusterRadius * (1 - relativeOverlap);
368 
369  // Step 3: using the maximum distance and the bounding box, compute the number of clusters in each direction
370  // we ceil the number of clusters in order to guarantee the desired overlap
371  std::array<unsigned int, 3> nClustersGlobal{1, 1, 1};
372  for (int d = 0; d < globalBB.getDimension(); ++d)
373  nClustersGlobal[d] = std::ceil(std::max(1., globalBB.getEdgeLength(d) / maximumCenterDistance));
374 
375  // Step 4: Determine the centers of the clusters
376  Vertices centers;
377  // Vector used to temporarily store each center coordinates
378  std::vector<double> centerCoords(inMesh->getDimensions());
379  // Vector storing the distances between the cluster in each direction
380  std::vector<double> distances(inMesh->getDimensions());
381  // Vector storing the starting coordinates in each direction
382  std::vector<double> start(inMesh->getDimensions());
383  // Fill the constant vectos, i.e., the distances and the start
384  for (unsigned int d = 0; d < distances.size(); ++d) {
385  // this distance calculation guarantees that we have at least the minimal specified overlap, as we ceil the division for the number of clusters
386  // as an alternative, once could use distance = const = maximumCenterDistance, which might lead to better results when projecting and removing duplicates (?)
387  distances[d] = globalBB.getEdgeLength(d) / (nClustersGlobal[d]);
388  start[d] = globalBB.minCorner()(d);
389  }
390  PRECICE_DEBUG("Distances: {}", distances);
391 
392  // Step 5: Take care of the starting layer: if we start the grid at the globalBB edge we have an additional layer in each direction
393  if (startGridAtEdge) {
394  for (int d = 0; d < inMesh->getDimensions(); ++d) {
395  if (globalBB.getEdgeLength(d) > math::NUMERICAL_ZERO_DIFFERENCE) {
396  nClustersGlobal[d] += 1;
397  }
398  }
399  } // If we don't start at the edge, we shift the starting layer into the middle by 0.5 * distance
400  else {
401  std::transform(start.begin(), start.end(), distances.begin(), start.begin(), [](auto s, auto d) { return s + 0.5 * d; });
402  }
403 
404  // Print some information
405  PRECICE_DEBUG("Global cluster distribution: {}", nClustersGlobal);
406  unsigned int nTotalClustersGlobal = std::accumulate(nClustersGlobal.begin(), nClustersGlobal.end(), 1U, std::multiplies<unsigned int>());
407  PRECICE_DEBUG("Global number of total clusters (tentative): {}", nTotalClustersGlobal);
408 
409  // Step 6: transform the global metrics (number of clusters and starting layer) into local ones
410  // Since the local and global bounding boxes are the same in the current implementation, the
411  // global metrics and the local metrics will be the same.
412  // First, we need to update the starting points of our clustering scheme
413  std::array<unsigned int, 3> nClustersLocal{1, 1, 1};
414  for (unsigned int d = 0; d < start.size(); ++d) {
415  // Exclude the case where we have no distances (nClustersGlobal[d] == 1 )
416  if (distances[d] > 0) {
417  start[d] += std::ceil(((localBB.minCorner()(d) - start[d]) / distances[d]) - math::NUMERICAL_ZERO_DIFFERENCE) * distances[d];
418  // One cluster for the starting point and a further one for each distance we can fit into the BB
419  nClustersLocal[d] = 1 + std::floor(((localBB.maxCorner()(d) - start[d]) / distances[d]) + math::NUMERICAL_ZERO_DIFFERENCE);
420  }
421  }
422 
423  unsigned int nTotalClustersLocal = std::accumulate(nClustersLocal.begin(), nClustersLocal.end(), 1U, std::multiplies<unsigned int>());
424  centers.resize(nTotalClustersLocal, mesh::Vertex({centerCoords, -1}));
425 
426  // Print some information
427  PRECICE_DEBUG("Local cluster distribution: {}", nClustersLocal);
428  PRECICE_DEBUG("Local number of total clusters (tentative): {}", nTotalClustersLocal);
429 
430  // Step 7: fill the center container using the zCurve for indexing. For the further processing, it is also important to ensure
431  // that the ID within the center Vertex class coincides with the position of the center vertex in the container.
432  // We start with the (bottom left) corner and iterate over all dimension
433  if (inMesh->getDimensions() == 2) {
434  centerCoords[0] = start[0];
435  for (unsigned int x = 0; x < nClustersLocal[0]; ++x, centerCoords[0] += distances[0]) {
436  centerCoords[1] = start[1];
437  for (unsigned int y = 0; y < nClustersLocal[1]; ++y, centerCoords[1] += distances[1]) {
438  auto id = zCurve(std::array<unsigned int, 3>{{x, y, 0}}, nClustersLocal);
439  PRECICE_ASSERT(id < static_cast<int>(centers.size()) && id >= 0);
440  centers[id] = mesh::Vertex({centerCoords, id});
441  }
442  }
443  } else {
444  centerCoords[0] = start[0];
445  for (unsigned int x = 0; x < nClustersLocal[0]; ++x, centerCoords[0] += distances[0]) {
446  centerCoords[1] = start[1];
447  for (unsigned int y = 0; y < nClustersLocal[1]; ++y, centerCoords[1] += distances[1]) {
448  centerCoords[2] = start[2];
449  for (unsigned int z = 0; z < nClustersLocal[2]; ++z, centerCoords[2] += distances[2]) {
450  auto id = zCurve(std::array<unsigned int, 3>{{x, y, z}}, nClustersLocal);
451  PRECICE_ASSERT(id < static_cast<int>(centers.size()) && id >= 0);
452  centers[id] = mesh::Vertex({centerCoords, id});
453  }
454  }
455  }
456  }
457 
458  // Step 8: tag vertices, which should be removed later on
459 
460  // Step 8a: tag empty clusters
461  tagEmptyClusters(centers, clusterRadius, inMesh);
462  tagEmptyClusters(centers, clusterRadius, outMesh);
463  PRECICE_DEBUG("Number of non-tagged centers after empty clusters were filtered: {}", std::count_if(centers.begin(), centers.end(), [](auto &v) { return !v.isTagged(); }));
464 
465  // Step 9: Move the cluster centers if desired
466  if (projectClustersToInput) {
467  projectClusterCentersToinputMesh(centers, inMesh);
468  // @todo: The duplication should probably be defined in terms of the target distance, not the actual distance. Find good default values
469  const auto duplicateThreshold = 0.4 * (*std::min_element(distances.begin(), distances.end()));
470  PRECICE_DEBUG("Tagging duplicates using the threshold value {} for the regular distances {}", duplicateThreshold, distances);
471  // usually inMesh->getDimensions() == 2
472  // to also cover the case where we have only a single layer in a 3D computation we use the number of clusters in z direction
473  // Step 8b or 9a: Moving cluster centers might lead to duplicate centers or centers being too close to each other. Therefore,
474  // we tag duplicate centers (see also the documentation of \ref tagDuplicateCenters ).
475  if (nClustersLocal[2] == 1) {
476  tagDuplicateCenters<2>(centers, nClustersLocal, duplicateThreshold);
477  } else {
478  tagDuplicateCenters<3>(centers, nClustersLocal, duplicateThreshold);
479  }
480  // the tagging here won't filter out a lot of clusters, but finding them anyway allows us to allocate the right amount of
481  // memory for the cluster vector in the mapping, which would otherwise be expensive
482  // we cannot hit any empty vertices in the inputmesh
483  tagEmptyClusters(centers, clusterRadius, outMesh);
484  PRECICE_DEBUG("Number of non-tagged centers after duplicate tagging : {}", std::count_if(centers.begin(), centers.end(), [](auto &v) { return !v.isTagged(); }));
485  }
486  PRECICE_ASSERT(std::all_of(centers.begin(), centers.end(), [idx = 0](auto &v) mutable { return v.getID() == idx++; }));
487 
488  // Step 10: finally, remove the vertices from the center container
489  removeTaggedVertices(centers);
490  PRECICE_ASSERT(std::none_of(centers.begin(), centers.end(), [](auto &v) { return v.isTagged(); }));
491  PRECICE_CHECK(centers.size() > 0, "Too many vertices have been filtered out.");
492 
493  return {clusterRadius, centers};
494 }
