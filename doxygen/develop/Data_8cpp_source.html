127  // Shrink Buffer
128  if (expectedSize < actualSize) {

131  // Enlarge Buffer

134  utils::append(_sample.values, Eigen::VectorXd(Eigen::VectorXd::Zero(leftToAllocate)));
135  }
136  PRECICE_DEBUG("Data {} now has {} values", _name, _sample.values.size());
