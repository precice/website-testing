137 void TestContext::initializeMPI(const TestContext::Participants &participants)
138 {
139  auto baseComm = Par::current();
140  const int globalRank = baseComm->rank();
141  const int available = baseComm->size();
142  const int required = std::accumulate(participants.begin(), participants.end(), 0, [](int total, const ParticipantState &next) { return total + next.size; });
143  if (required > available) {
144  throw std::runtime_error{"This test requests " + std::to_string(required) + " ranks, but there are only " + std::to_string(available) + " available"};
145  }
146 
147  // Restrict the communicator to the total required size
148  Par::restrictCommunicator(required);
149 
150  // Mark all unnecessary ranks as invalid and return
151  if (globalRank >= required) {
152  invalid = true;
153  return;
154  }
155 
156  // If there was only a single participant requested, then update its info and we are done.
157  if (participants.size() == 1) {
158  auto &participant = participants.front();
159  if (!invalid) {
160  setContextFrom(participant, globalRank);
161  }
162  return;
163  }
164 
165  // If there were multiple participants requested, we need to split the restricted comm
166  if (participants.size() > 1) {
167  int offset = 0;
168  for (const auto &participant : participants) {
169  const auto localRank = globalRank - offset;
170  // Check if my global rank maps to this participant
171  if (localRank < participant.size) {
172  Par::splitCommunicator(participant.name);
173  setContextFrom(participant, localRank);
174  return;
175  }
176  offset += participant.size;
177  }
178  }
179 }
