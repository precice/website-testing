54 Eigen::VectorXd Waveform::sample(double normalizedDt)
55 {
56  PRECICE_ASSERT(_storageIsInitialized);
57  PRECICE_ASSERT(normalizedDt >= 0, "Sampling outside of valid range!");
58  PRECICE_ASSERT(normalizedDt <= 1, "Sampling outside of valid range!");
59 
60  const int usedOrder = computeUsedOrder(_interpolationOrder, _numberOfStoredSamples);
61 
62  if (usedOrder == 0) {
63  // constant interpolation = just use sample at the end of the window: x(dt) = x^t
64  PRECICE_ASSERT(_numberOfStoredSamples > 0);
65  return this->_timeWindowsStorage.col(0);
66  }
67  Eigen::VectorXd interpolatedValue;
68  if (usedOrder == 1) {
69  // linear interpolation inside window: x(dt) = dt * x^t + (1-dt) * x^(t-1)
70  PRECICE_ASSERT(_numberOfStoredSamples > 1);
71  interpolatedValue = this->_timeWindowsStorage.col(0) * normalizedDt; // = dt * x^t
72  interpolatedValue += this->_timeWindowsStorage.col(1) * (1 - normalizedDt); // = dt * x^t + (1-dt) * x^(t-1)
73  return interpolatedValue;
74  }
75  PRECICE_ASSERT(usedOrder == 2);
76  // quadratic interpolation inside window: x(dt) = x^t * (dt^2 + dt)/2 + x^(t-1) * (1-dt^2)+ x^(t-2) * (dt^2-dt)/2
77  interpolatedValue = this->_timeWindowsStorage.col(0) * (normalizedDt + 1) * normalizedDt * 0.5;
78  interpolatedValue += this->_timeWindowsStorage.col(1) * (1 - normalizedDt * normalizedDt);
79  interpolatedValue += this->_timeWindowsStorage.col(2) * (normalizedDt - 1) * normalizedDt * 0.5;
80  return interpolatedValue;
81 }
