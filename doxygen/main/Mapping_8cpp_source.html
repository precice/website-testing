117 void Mapping::scaleConsistentMapping(int inputDataID, int outputDataID) const
118 {
119  // Only serial participant is supported for scale-consistent mapping
120  PRECICE_ASSERT((not utils::IntraComm::isPrimary()) and (not utils::IntraComm::isSecondary()));
121 
122  // If rank is not empty and do not contain connectivity information, raise error
123  if ((input()->edges().empty() and (not input()->vertices().empty())) or
124  (((input()->getDimensions() == 3) and input()->triangles().empty()) and (not input()->vertices().empty()))) {
125  logging::Logger _log{"mapping::Mapping"};
126  PRECICE_ERROR("Connectivity information is missing for the mesh {}. "
127  "Scaled consistent mapping requires connectivity information.",
128  input()->getName());
129  }
130  if ((output()->edges().empty() and (not output()->vertices().empty())) or
131  (((output()->getDimensions() == 3) and output()->triangles().empty()) and (not output()->vertices().empty()))) {
132  logging::Logger _log{"mapping::Mapping"};
133  PRECICE_ERROR("Connectivity information is missing for the mesh {}. "
134  "Scaled consistent mapping requires connectivity information.",
135  output()->getName());
136  }
137 
138  auto &outputValues = output()->data(outputDataID)->values();
139  int valueDimensions = input()->data(inputDataID)->getDimensions();
140 
141  // Integral is calculated on each direction separately
142  auto integralInput = mesh::integrate(input(), input()->data(inputDataID));
143  auto integralOutput = mesh::integrate(output(), output()->data(outputDataID));
144 
145  // Create reshape the output values vector to matrix
146  Eigen::Map<Eigen::MatrixXd> outputValuesMatrix(outputValues.data(), valueDimensions, outputValues.size() / valueDimensions);
147 
148  // Scale in each direction
149  Eigen::VectorXd scalingFactor = integralInput.array() / integralOutput.array();
150  outputValuesMatrix.array().colwise() *= scalingFactor.array();
