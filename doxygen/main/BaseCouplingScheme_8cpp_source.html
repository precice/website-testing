206 void BaseCouplingScheme::advance()
207 {
208  PRECICE_TRACE(_timeWindows, _time);
209  checkCompletenessRequiredActions();
210  PRECICE_ASSERT(_isInitialized, "Before calling advance() coupling scheme has to be initialized via initialize().");
211  _hasDataBeenReceived = false;
212  _isTimeWindowComplete = false;
213 
214  PRECICE_ASSERT(_couplingMode != Undefined);
215 
216  if (reachedEndOfTimeWindow()) {
217 
218  _timeWindows += 1; // increment window counter. If not converged, will be decremented again later.
219 
220  bool convergence = exchangeDataAndAccelerate();
221 
222  if (isImplicitCouplingScheme()) { // check convergence
223  if (not convergence) { // repeat window
224  PRECICE_DEBUG("No convergence achieved");
225  requireAction(constants::actionReadIterationCheckpoint());
226  // The computed time window part equals the time window size, since the
227  // time window remainder is zero. Subtract the time window size and do another
228  // coupling iteration.
229  PRECICE_ASSERT(math::greater(_computedTimeWindowPart, 0.0));
230  _time = _time - _computedTimeWindowPart;
231  _timeWindows -= 1;
232  } else { // write output, prepare for next window
233  PRECICE_DEBUG("Convergence achieved");
234  advanceTXTWriters();
235  PRECICE_INFO("Time window completed");
236  _isTimeWindowComplete = true;
237  if (isCouplingOngoing()) {
238  PRECICE_DEBUG("Setting require create checkpoint");
239  requireAction(constants::actionWriteIterationCheckpoint());
240  }
241  }
242  //update iterations
243  _totalIterations++;
244  if (not convergence) {
245  _iterations++;
246  } else {
247  _iterations = 1;
248  }
249  } else {
250  PRECICE_INFO("Time window completed");
251  _isTimeWindowComplete = true;
252  }
253  if (isCouplingOngoing()) {
254  PRECICE_ASSERT(_hasDataBeenReceived);
255  }
256  _computedTimeWindowPart = 0.0; // reset window
257  }

513 bool BaseCouplingScheme::measureConvergence()
514 {
515  PRECICE_TRACE();
516  PRECICE_ASSERT(not doesFirstStep());
517  bool allConverged = true;
518  bool oneSuffices = false; //at least one convergence measure suffices and did converge
519  bool oneStrict = false; //at least one convergence measure is strict and did not converge
520  PRECICE_ASSERT(_convergenceMeasures.size() > 0);
521  if (not utils::IntraComm::isSecondary()) {
522  _convergenceWriter->writeData("TimeWindow", _timeWindows - 1);
523  _convergenceWriter->writeData("Iteration", _iterations);
524  }
525  for (const auto &convMeasure : _convergenceMeasures) {
526  PRECICE_ASSERT(convMeasure.couplingData != nullptr);
527  PRECICE_ASSERT(convMeasure.measure.get() != nullptr);
528 
529  convMeasure.measure->measure(convMeasure.couplingData->previousIteration(), convMeasure.couplingData->values());
530 
531  if (not utils::IntraComm::isSecondary() && convMeasure.doesLogging) {
532  _convergenceWriter->writeData(convMeasure.logHeader(), convMeasure.measure->getNormResidual());
533  }
534 
535  if (not convMeasure.measure->isConvergence()) {
536  allConverged = false;
537  if (convMeasure.strict) {
538  oneStrict = true;
539  PRECICE_CHECK(_iterations < _maxIterations,
540  "The strict convergence measure for data \"" + convMeasure.couplingData->getDataName() +
541  "\" did not converge within the maximum allowed iterations, which terminates the simulation. "
542  "To avoid this forced termination do not mark the convergence measure as strict.")
543  }
544  } else if (convMeasure.suffices == true) {
545  oneSuffices = true;
546  }
547 
548  PRECICE_INFO(convMeasure.measure->printState(convMeasure.couplingData->getDataName()));
549  }
550 
551  if (allConverged) {
552  PRECICE_INFO("All converged");
553  } else if (oneSuffices && not oneStrict) { //strict overrules suffices
554  PRECICE_INFO("Sufficient measures converged");
555  }
556 
557  return allConverged || (oneSuffices && not oneStrict);
558 }

648 bool BaseCouplingScheme::doImplicitStep()
649 {
650  storeExtrapolationData();
651 
652  PRECICE_DEBUG("measure convergence of the coupling iteration");
653  bool convergence = measureConvergence();
654  // Stop, when maximal iteration count (given in config) is reached
655  if (_iterations == _maxIterations)
656  convergence = true;
657 
658  // coupling iteration converged for current time window. Advance in time.
659  if (convergence) {
660  if (_acceleration) {
661  _acceleration->iterationsConverged(getAccelerationData());
662  }
663  newConvergenceMeasurements();
664  } else {
665  // no convergence achieved for the coupling iteration within the current time window
666  if (_acceleration) {
