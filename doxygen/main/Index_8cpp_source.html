292 ProjectionMatch Index::findEdgeProjection(const Eigen::VectorXd &location, int n)
293 {
294  std::vector<ProjectionMatch> candidates;
295  candidates.reserve(n);
296  for (const auto &match : getClosestEdges(location, n)) {
297  auto polation = mapping::Polation(location, _mesh->edges()[match.index]);
298  if (polation.isInterpolation()) {
299  candidates.emplace_back(std::move(polation));
300  }
301  }
302  // Pick the smallest candidate by distance
303  auto min = std::min_element(candidates.begin(), candidates.end());
304  if (min != candidates.end()) {
305  return *min;
306  }
307  // Could not find edge projection element, fall back to vertex projection
308  return findVertexProjection(location);
309 }
