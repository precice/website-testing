156 {
157  PRECICE_TRACE(_name);

159 
160  const auto position = aAttributes.find(getName());
161  if (position == aAttributes.end()) {
162  if (not _hasDefaultValue) {
163  PRECICE_ERROR("Attribute \"{}\" is required, but was not defined.", _name);
164  }
165  set(_value, _defaultValue);
166  } else {
167  try {
168  readValueSpecific(position->second, _value);
169  } catch (const std::exception &e) {
170  PRECICE_ERROR(e.what());
171  }
172  if (_hasValidation) {

174  std::ostringstream stream;

191 }
