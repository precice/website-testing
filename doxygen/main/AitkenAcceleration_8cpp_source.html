51 void AitkenAcceleration::performAcceleration(
52  const DataMap &cplData)
53 {
54  PRECICE_TRACE();
55 
56  // Compute aitken relaxation factor
57  PRECICE_ASSERT(utils::contained(*_dataIDs.begin(), cplData));
58 
59  Eigen::VectorXd values;
60  Eigen::VectorXd oldValues;
61  for (int id : _dataIDs) {
62  utils::append(values, cplData.at(id)->values());
63  utils::append(oldValues, Eigen::VectorXd(cplData.at(id)->previousIteration()));
64  }
65 
66  // Compute current residuals
67  Eigen::VectorXd residuals = values;
68  residuals -= oldValues;
69 
70  // Compute residual deltas and temporarily store it in _residuals
71  Eigen::VectorXd residualDeltas = _residuals;
72  residualDeltas *= -1.0;
73  residualDeltas += residuals;
74 
75  // Select/compute aitken factor depending on current iteration count
76  if (_iterationCounter == 0) {
77  _aitkenFactor = math::sign(_aitkenFactor) * std::min(_initialRelaxation, std::abs(_aitkenFactor));
78  } else {
79  // compute fraction of aitken factor with residuals and residual deltas
80  double nominator = utils::IntraComm::dot(_residuals, residualDeltas);
81  double denominator = utils::IntraComm::dot(residualDeltas, residualDeltas);
82  _aitkenFactor = -_aitkenFactor * (nominator / denominator);
83  }
84 
85  PRECICE_DEBUG("AitkenFactor: {}", _aitkenFactor);
86 
87  // Perform relaxation with aitken factor
88  double omega = _aitkenFactor;
89  double oneMinusOmega = 1.0 - omega;
90  for (const DataMap::value_type &pair : cplData) {
91  auto & values = pair.second->values();
92  const auto &oldValues = pair.second->previousIteration();
93  values *= omega;
94  for (int i = 0; i < values.size(); i++) {
95  values(i) += oldValues(i) * oneMinusOmega;
96  }
97  }
98

100  _residuals = residuals;
101

103 }
