64 void AitkenAcceleration::performAcceleration(
65  DataMap &cplData)
66 {
67  PRECICE_TRACE();
68 
69  // Compute aitken relaxation factor
70  PRECICE_ASSERT(utils::contained(*_dataIDs.begin(), cplData));
71 
72  concatenateCouplingData(cplData, _dataIDs, _values, _oldValues);
73 
74  // Compute current residual = values - oldValues
75  Eigen::VectorXd residuals = _values - _oldValues;
76 
77  // Compute residual deltas (= residuals - oldResiduals) and store it in _oldResiduals
78  Eigen::VectorXd residualDeltas = residuals - _oldResiduals;
79 
80  // We need to update the preconditioner in every iteration
81  if (_dataIDs.size() > 1) {
82  _preconditioner->update(false, _values, residuals);
83  }
84 
85  // Select/compute aitken factor depending on current iteration count
86  if (_iterationCounter == 0) {
87  // preconditioner not necessary
88  _aitkenFactor = math::sign(_aitkenFactor) * std::min(_initialRelaxation, std::abs(_aitkenFactor));
89  } else {
90  // If we have more than one data set, we scale the data to get a better approximation
91  // of the Aitken factor
92  if (_dataIDs.size() > 1) {
93  _preconditioner->apply(residualDeltas);
94  _preconditioner->apply(_oldResiduals);
95  }
96  // compute fraction of aitken factor with residuals and residual deltas
97  double nominator = utils::IntraComm::dot(_oldResiduals, residualDeltas);
98  double denominator = utils::IntraComm::dot(residualDeltas, residualDeltas);
99  _aitkenFactor = -_aitkenFactor * (nominator / denominator);
100  }
101 
102  PRECICE_DEBUG("AitkenFactor: {}", _aitkenFactor);
103 
104  // Perform relaxation with aitken factor
105  applyRelaxation(_aitkenFactor, cplData);
106 
107  // Store residuals for next iteration
108  _oldResiduals = std::move(residuals);
109 
110  _iterationCounter++;
111 }
