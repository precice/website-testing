402 void EventRegistry::collect()
403 {
404 #ifndef PRECICE_NO_MPI
405  // Register MPI datatype
406  MPI_Datatype MPI_EVENTDATA;
407  int blocklengths[] = {255, 1, 3, 2};
408  MPI_Aint displacements[] = {offsetof(MPI_EventData, name), offsetof(MPI_EventData, count),
409  offsetof(MPI_EventData, total), offsetof(MPI_EventData, dataSize)};
410  MPI_Datatype types[] = {MPI_CHAR, MPI_INT, MPI_LONG, MPI_INT};
411  MPI_Type_create_struct(4, blocklengths, displacements, types, &MPI_EVENTDATA);
412  MPI_Type_commit(&MPI_EVENTDATA);
413 
414  int rank, MPIsize;
415  MPI_Comm_rank(comm, &rank);
416  MPI_Comm_size(comm, &MPIsize);
417 
418  std::vector<MPI_Request> requests;
419  std::vector<int> eventsPerRank(MPIsize);
420  size_t eventsSize = localRankData.evData.size();
421  MPI_Gather(&eventsSize, 1, MPI_INT, eventsPerRank.data(), 1, MPI_INT, 0, comm);
422 
423  std::vector<MPI_EventData> eventSendBuf(eventsSize);
424  std::vector<std::vector<long>> stateChangesBuf(eventsSize);
425  int i = 0;
426 
427  MPI_Request req;
428 
429  // Send the times from the local RankData
430  std::array<long, 2> times = {localRankData.initializedAt.time_since_epoch().count(),
431  localRankData.finalizedAt.time_since_epoch().count()};
432  MPI_Isend(&times, times.size(), MPI_LONG, 0, 0, comm, &req);
433  requests.push_back(req);
434 
435  // Send all events from all ranks, including rank 0, to rank 0
436  for (auto const &evData : localRankData.evData) {
437  const auto & ev = evData.second;
438  MPI_EventData eventdata;
439 
440  // Send aggregated EventData
441  assert(evData.first.size() <= sizeof(eventdata.name));
442  evData.first.copy(eventSendBuf[i].name, sizeof(eventdata.name));
443  eventSendBuf[i].count = ev.getCount();
444  eventSendBuf[i].total = ev.getTotal();
445  eventSendBuf[i].max = ev.getMax();
446  eventSendBuf[i].min = ev.getMin();
447  eventSendBuf[i].dataSize = ev.getData().size();
448  eventSendBuf[i].stateChangesSize = ev.stateChanges.size();
449  MPI_Isend(&eventSendBuf[i], 1, MPI_EVENTDATA, 0, 0, comm, &req);
450  requests.push_back(req);
451 
452  // Send the state changes
453  for (auto const &sc : ev.stateChanges) {
454  stateChangesBuf[i].push_back(static_cast<long>(sc.first)); // state
455  stateChangesBuf[i].push_back(
456  std::chrono::duration_cast<std::chrono::milliseconds>(sc.second.time_since_epoch()).count());
457  }
458  MPI_Isend(stateChangesBuf[i].data(), ev.stateChanges.size() * 2, MPI_LONG, 0, 0, comm, &req);
459  requests.push_back(req);
460 
461  // Send the map that stores the data associated with an event
462  for (auto const &md : ev.getData()) {
463  auto &key = std::get<0>(md);
464  auto &val = std::get<1>(md);
465  MPI_Isend(const_cast<char *>(key.c_str()), key.size(), MPI_CHAR, 0, 0, comm, &req);
466  requests.push_back(req);
467  MPI_Isend(const_cast<int *>(val.data()), val.size(), MPI_INT, 0, 0, comm, &req);
468  requests.push_back(req);
469  }
470 
471  ++i;
472  }
473 
474  // Receive
475  if (rank == 0) {
476  for (int i = 0; i < MPIsize; ++i) {
477  RankData data;
478  // Receive initialized and finalized times
479  std::array<long, 2> recvTimes;
480  MPI_Recv(&recvTimes, 2, MPI_LONG, i, MPI_ANY_TAG, comm, MPI_STATUS_IGNORE);
481  data.initializedAt = sys_clk::time_point(sys_clk::duration(recvTimes[0]));
482  data.finalizedAt = sys_clk::time_point(sys_clk::duration(recvTimes[1]));
483 
484  // Receive all events from this rank
485  for (int j = 0; j < eventsPerRank[i]; ++j) {
486  // Receive aggregated EventData
487  MPI_EventData ev;
488  MPI_Recv(&ev, 1, MPI_EVENTDATA, i, MPI_ANY_TAG, comm, MPI_STATUS_IGNORE);
489 
490  // Receive all state changes for this event
491  std::vector<long> recvStateChanges(ev.stateChangesSize * 2);
492  MPI_Recv(recvStateChanges.data(), recvStateChanges.size(),
493  MPI_LONG, i, MPI_ANY_TAG, comm, MPI_STATUS_IGNORE);
494  Event::StateChanges stateChanges; // evtl. reserve
495  for (size_t k = 0; k < recvStateChanges.size(); k += 2) {
496  stateChanges.emplace_back(static_cast<Event::State>(recvStateChanges[k]),
497  stdy_clk::time_point(std::chrono::milliseconds(recvStateChanges[k + 1])));
498  }
499 
500  // Receive the map that stores the data associated with an event
501  Event::Data dataMap;
502  for (int j = 0; j < ev.dataSize; j++) {
503  MPI_Status status;
504  int count = 0;
505  MPI_Probe(i, MPI_ANY_TAG, comm, &status);
506  MPI_Get_count(&status, MPI_CHAR, &count);
507  std::string key(count, '\0');
508  MPI_Recv(&key[0], count, MPI_CHAR, i, MPI_ANY_TAG, comm, MPI_STATUS_IGNORE);
509  MPI_Probe(i, MPI_ANY_TAG, comm, &status);
510  MPI_Get_count(&status, MPI_INT, &count);
511  std::vector<int> val(count);
512  MPI_Recv(val.data(), count, MPI_INT, i, MPI_ANY_TAG, comm, MPI_STATUS_IGNORE);
513  dataMap[key] = val;
514  }
515 
516  // Create the EventData
517  EventData ed(ev.name, ev.count, ev.total, ev.max, ev.min, dataMap, stateChanges);
518  data.addEventData(std::move(ed));
519  }
520  globalRankData.push_back(data);
521  }
522  }
523  MPI_Waitall(requests.size(), requests.data(), MPI_STATUSES_IGNORE);
524  MPI_Type_free(&MPI_EVENTDATA);
525 #else
526  globalRankData.push_back(localRankData);
527 #endif
528 }

569 {

571  auto first = std::min_element(std::begin(globalRankData), std::end(globalRankData),
