292 void Mesh::addMesh(
293  Mesh &deltaMesh)
294 {
295  PRECICE_TRACE();
296  PRECICE_ASSERT(_dimensions == deltaMesh.getDimensions());
297 
298  boost::container::flat_map<VertexID, Vertex *> vertexMap;
299  vertexMap.reserve(deltaMesh.vertices().size());
300  Eigen::VectorXd coords(_dimensions);
301  for (const Vertex &vertex : deltaMesh.vertices()) {
302  coords = vertex.getCoords();
303  Vertex &v = createVertex(coords);
304  v.setGlobalIndex(vertex.getGlobalIndex());
305  if (vertex.isTagged())
306  v.tag();
307  v.setOwner(vertex.isOwner());
308  PRECICE_ASSERT(vertex.getID() >= 0, vertex.getID());
309  vertexMap[vertex.getID()] = &v;
310  }
311 
312  // you cannot just take the vertices from the edge and add them,
313  // since you need the vertices from the new mesh
314  // (which may differ in IDs)
315  for (const Edge &edge : deltaMesh.edges()) {
316  VertexID vertexIndex1 = edge.vertex(0).getID();
317  VertexID vertexIndex2 = edge.vertex(1).getID();
318  PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
319  (vertexMap.count(vertexIndex2) == 1));
320  createEdge(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2]);
321  }
322 
323  for (const Triangle &triangle : deltaMesh.triangles()) {
324  VertexID vertexIndex1 = triangle.vertex(0).getID();
325  VertexID vertexIndex2 = triangle.vertex(1).getID();
326  VertexID vertexIndex3 = triangle.vertex(2).getID();
327  PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
328  (vertexMap.count(vertexIndex2) == 1) &&
329  (vertexMap.count(vertexIndex3) == 1));
330  createTriangle(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2], *vertexMap[vertexIndex3]);
331  }
332 
333  for (const Tetrahedron &tetra : deltaMesh.tetrahedra()) {
334  VertexID vertexIndex1 = tetra.vertex(0).getID();
335  VertexID vertexIndex2 = tetra.vertex(1).getID();
336  VertexID vertexIndex3 = tetra.vertex(2).getID();
337  VertexID vertexIndex4 = tetra.vertex(3).getID();
338 
339  PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
340  (vertexMap.count(vertexIndex2) == 1) &&
341  (vertexMap.count(vertexIndex3) == 1) &&
342  (vertexMap.count(vertexIndex4) == 1));
343  createTetrahedron(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2], *vertexMap[vertexIndex3], *vertexMap[vertexIndex4]);
344  }
345  _index.clear();
346 }

364 void Mesh::removeDuplicates()
365 {
366  // Remove duplicate tetrahedra
367  auto tetrahedraCnt = _tetrahedra.size();
368  std::sort(_tetrahedra.begin(), _tetrahedra.end());
369  auto lastTetrahedron = std::unique(_tetrahedra.begin(), _tetrahedra.end());
370  _tetrahedra = TetraContainer{_tetrahedra.begin(), lastTetrahedron};
371 
372  // Remove duplicate triangles
373  auto triangleCnt = _triangles.size();
374  std::sort(_triangles.begin(), _triangles.end());
375  auto lastTriangle = std::unique(_triangles.begin(), _triangles.end());
376  _triangles = TriangleContainer{_triangles.begin(), lastTriangle};
377 
378  // Remove duplicate edges
379  auto edgeCnt = _edges.size();
380  std::sort(_edges.begin(), _edges.end());
381  auto lastEdge = std::unique(_edges.begin(), _edges.end());
382  _edges = EdgeContainer{_edges.begin(), lastEdge};
383 
384  PRECICE_DEBUG("Compression removed {} tetrahedra ({} to {}), {} triangles ({} to {}), and {} edges ({} to {})",
385  tetrahedraCnt - _tetrahedra.size(), tetrahedraCnt, _tetrahedra.size(),
386  triangleCnt - _triangles.size(), triangleCnt, _triangles.size(),
387  edgeCnt - _edges.size(), edgeCnt, _edges.size());
388 }

415 void Mesh::generateImplictPrimitives()
416 {
417  if (_triangles.empty() && _tetrahedra.empty()) {
418  PRECICE_DEBUG("No implicit primitives required");
419  return; // no implicit primitives
420  }
421 
422  // count explicit primitives for debug
423  const auto explTriangles = _triangles.size();
424  const auto explEdges = _triangles.size();
425 
426  // First handle all explicit tetrahedra
427 
428  // Build a set of all explicit triangles
429  using ExisitingTriangle = std::tuple<Vertex *, Vertex *, Vertex *>;
430  std::set<ExisitingTriangle> triangles;
431  for (auto &t : _triangles) {
432  triangles.insert(sortedVertexPtrsFor(t));
433  }
434 
435  // Generate all missing implicit triangles of explicit tetrahedra
436  // Update the triangles set used by the implicit edge generation
437  auto createTriangleIfMissing = [&](Vertex *a, Vertex *b, Vertex *c) {
438  if (triangles.count({a, b, c}) == 0) {
439  triangles.emplace(a, b, c);
440  createTriangle(*a, *b, *c);
441  };
442  };
443  for (auto &t : _tetrahedra) {
444  auto [a, b, c, d] = sortedVertexPtrsFor(t);
445  // Make sure these are in the same order as above
446  createTriangleIfMissing(a, b, c);
447  createTriangleIfMissing(a, b, d);
448  createTriangleIfMissing(a, c, d);
449  createTriangleIfMissing(b, c, d);
450  }
451 
452  // Second handle all triangles, both explicit and implicit from the tetrahedron phase
453  // Build an set of all explicit triangles
454  using ExisitingEdge = std::tuple<Vertex *, Vertex *>;
455  std::set<ExisitingEdge> edges;
456  for (auto &e : _edges) {
457  edges.emplace(sortedVertexPtrsFor(e));
458  }
459 
460  // generate all missing implicit edges of implicit and explicit triangles
461  auto createEdgeIfMissing = [&](Vertex *a, Vertex *b) {
462  if (edges.count({a, b}) == 0) {
463  edges.emplace(a, b);
464  createEdge(*a, *b);
465  };
466  };
467  for (auto &t : _triangles) {
468  auto [a, b, c] = sortedVertexPtrsFor(t);
469  // Make sure these are in the same order as above
470  createEdgeIfMissing(a, b);
471  createEdgeIfMissing(a, c);
472  createEdgeIfMissing(b, c);
473  }
474 
475  PRECICE_DEBUG("Generated {} implicit triangles and {} implicit edges",
476  _triangles.size() - explTriangles,
477  _edges.size() - explEdges);
478 }
