107 void QRFactorization::applyFilter(double singularityLimit, std::vector<int> &delIndices, Eigen::MatrixXd &V)
108 {
109  PRECICE_TRACE();
110  delIndices.resize(0);
111  if (_filter == Acceleration::QR1FILTER || _filter == Acceleration::QR1FILTER_ABS) {
112  bool linearDependence = true;
113  std::vector<int> delFlag(_cols, 0);
114  int delCols = 0;
115  while (linearDependence) {
116  linearDependence = false;
117  int index = 0; // actual index of checked column, \in [0, _cols] and _cols is decreasing
118  if (_cols > 1) {
119  for (size_t i = 0; i < delFlag.size(); i++) {
120  // index is not incremented, if columns has been deleted in previous rounds
121  if (delFlag[i] > 0)
122  continue;
123 
124  // QR1-filter
125  if (index >= cols())
126  break;
127  PRECICE_ASSERT(index < _cols, index, _cols);
128  double factor = (_filter == Acceleration::QR1FILTER_ABS) ? 1.0 : _R.norm();
129  if (std::fabs(_R(index, index)) < singularityLimit * factor) {
130 
131  linearDependence = true;
132  deleteColumn(index);
133  delFlag[i]++;
134  delIndices.push_back(i);
135  delCols++;
136  //break;
137  index--; // check same column index, as cols are shifted left
138  }
139  PRECICE_ASSERT(delCols + _cols == (int) delFlag.size(), (delCols + _cols), delFlag.size());
140  index++;
141  }
142  }
143  }
144  } else if (_filter == Acceleration::QR2FILTER) {
145  _Q.resize(0, 0);
146  _R.resize(0, 0);
147  _cols = 0;
148  _rows = V.rows();
149  // starting with the most recent input/output information, i.e., the latest column
150  // which is at position 0 in _matrixV (latest information is never filtered out!)
151  for (int k = 0; k < V.cols(); k++) {
152  Eigen::VectorXd v = V.col(k);
153  // this is the same as pushBack(v) as _cols grows within the insertion process
154  bool inserted = insertColumn(_cols, v, singularityLimit);
155  if (!inserted) {
156  delIndices.push_back(k);
157  }
158  }
159  }
160  std::sort(delIndices.begin(), delIndices.end());

210 {
211  PRECICE_TRACE(k);
212 
213  Eigen::VectorXd v(vec);
214 
215  if (_cols == 0)
216  _rows = v.size();
217 
218  bool applyFilter = (singularityLimit > 0.0);
219 
220  PRECICE_ASSERT(k >= 0, k);
221  PRECICE_ASSERT(k <= _cols, k, _cols);
222  PRECICE_ASSERT(v.size() == _rows, v.size(), _rows);
223 
224  _cols++;
225 
226  // orthogonalize v to columns of Q
227  Eigen::VectorXd u(_cols);
228  double rho_orth = 0., rho0 = 0.;
229  if (applyFilter)
230  rho0 = utils::IntraComm::l2norm(v);
231 
232  int err = orthogonalize(v, u, rho_orth, _cols - 1);
233 
234  // on of the following is true
235  // - either ||v_orth|| / ||v|| <= 0.7 was true and the re-orthogonalization process failed 4 times
236  // - or the system is quadratic, thus no further column can be orthogonalized (||v_orth|| = rho_orth = 0)
237  // - or ||v_orth|| = rho_orth extremely small (almost zero), thus the column v is not very orthogonal to Q, discard
238  if (rho_orth <= std::numeric_limits<double>::min() || err < 0) {
239  PRECICE_DEBUG("The ratio ||v_orth|| / ||v|| is extremely small and either the orthogonalization process of column v failed or the system is quadratic.");
240 
241  // necessary for applyFilter with the QR-2 filer. In this case, the new column is not inserted, but discarded.
242  _cols--;
243  return false;
244  }
245 
246  // QR2-filter based on the new information added to the orthogonal system.
247  // if the new column incorporates less new information to the system than a
248  // prescribed threshold, the column is discarded
249  // rho_orth: the norm of the orthogonalized (but not normalized) column
250  // rho0: the norm of the initial column that is to be inserted
251  if (applyFilter && (rho0 * singularityLimit > rho_orth)) {
252  PRECICE_DEBUG("discarding column as it is filtered out by the QR2-filter: rho0*eps > rho_orth: {} > {}", rho0 * singularityLimit, rho_orth);
253  _cols--;
254  return false;
255  }
256 
257  // resize R(1:m, 1:m) -> R(1:m+1, 1:m+1)
258  _R.conservativeResize(_cols, _cols);
259  _R.col(_cols - 1) = Eigen::VectorXd::Zero(_cols);
260  _R.row(_cols - 1) = Eigen::VectorXd::Zero(_cols);
261 
262  for (int j = _cols - 2; j >= k; j--) {
263  for (int i = 0; i <= j; i++) {
264  _R(i, j + 1) = _R(i, j);
265  }
266  }
267 
268  for (int j = k + 1; j < _cols; j++) {
269  _R(j, j) = 0.;
270  }
271 
272  PRECICE_ASSERT(_R.cols() == _cols, _R.cols(), _cols);
273  //PRECICE_ASSERT(_R.rows() == _cols, _R.rows(), _cols);
274 
275  // resize Q(1:n, 1:m) -> Q(1:n, 1:m+1)
276  _Q.conservativeResize(_rows, _cols);
277  _Q.col(_cols - 1) = v;
278 
279  PRECICE_ASSERT(_Q.cols() == _cols, _Q.cols(), _cols);
280  PRECICE_ASSERT(_Q.rows() == _rows, _Q.rows(), _rows);
281 
282  // maintain decomposition and orthogonalization by application of givens rotations
283  for (int l = _cols - 2; l >= k; l--) {
284  QRFactorization::givensRot grot;
285  computeReflector(grot, u(l), u(l + 1));
286  Eigen::VectorXd Rr1 = _R.row(l);
287  Eigen::VectorXd Rr2 = _R.row(l + 1);
288  applyReflector(grot, l + 1, _cols, Rr1, Rr2);
289  _R.row(l) = Rr1;
290  _R.row(l + 1) = Rr2;
291  Eigen::VectorXd Qc1 = _Q.col(l);
292  Eigen::VectorXd Qc2 = _Q.col(l + 1);
293  applyReflector(grot, 0, _rows, Qc1, Qc2);
294  _Q.col(l) = Qc1;
295  _Q.col(l + 1) = Qc2;
296  }
297  for (int i = 0; i <= k; i++) {
298  _R(i, k) = u(i);
299  }
300 
301  return true;

315 int QRFactorization::orthogonalize(
316  Eigen::VectorXd &v,
317  Eigen::VectorXd &r,
318  double & rho,
319  int colNum)
320 {
321  PRECICE_TRACE();
322 
323  if (!utils::IntraComm::isParallel()) {
324  PRECICE_ASSERT(_globalRows == _rows, _globalRows, _rows);
325  }
326 
327  bool null = false;
328  bool termination = false;
329  double rho0 = 0., rho1 = 0.;
330  Eigen::VectorXd u = Eigen::VectorXd::Zero(_rows);
331  Eigen::VectorXd s = Eigen::VectorXd::Zero(colNum);
332  r = Eigen::VectorXd::Zero(_cols);
333 
334  rho = utils::IntraComm::l2norm(v); // distributed l2norm
335  rho0 = rho;
336  int k = 0;
337  while (!termination) {
338 
339  // take a gram-schmidt iteration
340  u = Eigen::VectorXd::Zero(_rows);
341  for (int j = 0; j < colNum; j++) {
342 
343  // dot-product <_Q(:,j), v >
344  Eigen::VectorXd Qc = _Q.col(j);
345 
346  // dot product <_Q(:,j), v> =: r_ij
347  double r_ij = utils::IntraComm::dot(Qc, v);
348  // save r_ij in s(j) = column of R
349  s(j) = r_ij;
350  // u is the sum of projections r_ij * _Q(:,j) = _Q(:,j) * <_Q(:,j), v>
351  u += _Q.col(j) * r_ij;
352  }
353  // add the furier coefficients over all orthogonalize iterations
354  for (int j = 0; j < colNum; j++) {
355  r(j) = r(j) + s(j);
356  }
357  // subtract projections from v, v is now orthogonal to columns of _Q
358  for (int i = 0; i < _rows; i++) {
359  v(i) = v(i) - u(i);
360  }
361  // rho1 = norm of orthogonalized new column v_tilde (though not normalized)
362  rho1 = utils::IntraComm::l2norm(v); // distributed l2norm
363 
364  // t = norm of r_(:,j) with j = colNum-1
365  double norm_coefficients = utils::IntraComm::l2norm(s); // distributed l2norm
366  k++;
367 
368  // treat the special case m=n
369  // Attention (intra-participant communication): Here, we need to compare the global _rows with colNum and NOT the local
370  // rows on the processor.
371  if (_globalRows == colNum) {
372  PRECICE_WARN("The least-squares system matrix is quadratic, i.e., the new column cannot be orthogonalized (and thus inserted) to the LS-system.\nOld columns need to be removed.");
373  v = Eigen::VectorXd::Zero(_rows);
374  rho = 0.;
375  return k;
376  }
377 
378  // take correct action if v_orth is null
379  if (rho1 <= std::numeric_limits<double>::min()) {
380  PRECICE_DEBUG("The norm of v_orthogonal is almost zero, i.e., failed to orthogonalize column v; discard.");
381  null = true;
382  rho1 = 1;
383  termination = true;
384  }
385 
398  if (rho1 * _theta <= rho0 + _omega * norm_coefficients) {
399  // exit to fail if too many iterations
400  if (k >= 4) {
401  PRECICE_WARN("Matrix Q is not sufficiently orthogonal. Failed to rorthogonalize new column after 4 iterations. New column will be discarded. The least-squares system is very bad conditioned and the quasi-Newton will most probably fail to converge.");
402  return -1;
403  }
404  rho0 = rho1;
405 
406  // termination, i.e., (rho0 + _omega * t < _theta *rho1)
407  } else {
408  termination = true;
409  }
410  }
411 
412  // normalize v
413  v /= rho1;
414  rho = null ? 0 : rho1;
415  r(colNum) = rho;
416  return k;
417 }

618 void QRFactorization::computeReflector(
619  QRFactorization::givensRot &grot,
620  double & x,
621  double & y)
622 {
623  double u = x;
624  double v = y;
625  if (v == 0) {
626  grot.sigma = 0;
627  grot.gamma = 1;
628  } else {
629  double mu = std::max(std::fabs(u), std::fabs(v));
630  double t = mu * std::sqrt(std::pow(u / mu, 2) + std::pow(v / mu, 2));
631  t *= (u < 0) ? -1 : 1;
632  grot.gamma = u / t;
633  grot.sigma = v / t;
634  x = t;
635  y = 0;
636  }
637 }

643 void QRFactorization::applyReflector(
644  const QRFactorization::givensRot &grot,
645  int k,
646  int l,
647  Eigen::VectorXd & p,
648  Eigen::VectorXd & q)
649 {
650  double nu = grot.sigma / (1. + grot.gamma);
651  for (int j = k; j < l; j++) {
652  double u = p(j);
653  double v = q(j);
654  double t = u * grot.gamma + v * grot.sigma;
655  p(j) = t;
656  q(j) = (t + u) * nu - v;
657  }
