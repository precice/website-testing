63 inline double computeSquaredDifference(
64  const std::array<double, 3> &u,
65  std::array<double, 3> v,
66  const std::array<bool, 3> & activeAxis)
67 {
68  // Subtract the values and multiply out dead dimensions
69  for (unsigned int d = 0; d < v.size(); ++d) {
70  v[d] = (u[d] - v[d]) * static_cast<int>(activeAxis[d]);
71  }
72  // @todo: this can be replaced by std::hypot when moving to C++17
73  return std::accumulate(v.begin(), v.end(), static_cast<double>(0.), [](auto &res, auto &val) { return res + val * val; });
74 }

77 inline void fillPolynomialEntries(Eigen::MatrixXd &matrix, const mesh::Mesh &mesh, Eigen::Index startIndex, std::array<bool, 3> activeAxis)
78 {
79  // Loop over all vertices in the mesh
80  for (auto i : boost::irange<Eigen::Index>(0, mesh.vertices().size())) {
81 
82  // 1. the constant contribution
83  matrix(i, startIndex) = 1.0;
84 
85  // 2. the linear contribution
86  const auto & u = mesh.vertices()[i].rawCoords();
87  unsigned int k = 0;
88  // Loop over all three space dimension and ignore dead axis
89  for (unsigned int d = 0; d < activeAxis.size(); ++d) {
90  if (activeAxis[d]) {
91  PRECICE_ASSERT(matrix.rows() > i, matrix.rows(), i);
92  PRECICE_ASSERT(matrix.cols() > startIndex + 1 + k, matrix.cols(), startIndex + 1 + k);
93  matrix(i, startIndex + 1 + k) = u[d];
94  ++k;
95  }
96  }
97  }

101 Eigen::MatrixXd buildMatrixCLU(RADIAL_BASIS_FUNCTION_T basisFunction, const mesh::Mesh &inputMesh, std::array<bool, 3> activeAxis, Polynomial polynomial)
102 {
103  // Treat the 2D case as 3D case with dead axis
104  const unsigned int deadDimensions = std::count(activeAxis.begin(), activeAxis.end(), false);
105  const unsigned int dimensions = 3;
106  const unsigned int polyparams = polynomial == Polynomial::ON ? 1 + dimensions - deadDimensions : 0;
107 
108  // Add linear polynom degrees if polynomial requires this
109  const auto inputSize = inputMesh.vertices().size();
110  const auto n = inputSize + polyparams;
111 
112  PRECICE_ASSERT((inputMesh.getDimensions() == 3) || activeAxis[2] == false);
113  PRECICE_ASSERT((inputSize >= 1 + polyparams) || polynomial != Polynomial::ON, inputSize);
114 
115  Eigen::MatrixXd matrixCLU(n, n);
116  matrixCLU.setZero();
117 
118  // Compute RBF matrix entries
119  for (auto i : boost::irange<Eigen::Index>(0, inputSize)) {
120  for (auto j : boost::irange<Eigen::Index>(i, inputSize)) {
121  const auto &u = inputMesh.vertices()[i].rawCoords();
122  const auto &v = inputMesh.vertices()[j].rawCoords();
123  double squaredDifference = computeSquaredDifference(u, v, activeAxis);
124  matrixCLU(i, j) = basisFunction.evaluate(std::sqrt(squaredDifference));
125  }
126  }
127 
128  // Add potentially the polynomial contribution in the matrix
129  if (polynomial == Polynomial::ON) {
130  fillPolynomialEntries(matrixCLU, inputMesh, inputSize, activeAxis);
131  }
132  matrixCLU.triangularView<Eigen::Lower>() = matrixCLU.transpose();
133  return matrixCLU;
134 }

172 RadialBasisFctSolver::RadialBasisFctSolver(RADIAL_BASIS_FUNCTION_T basisFunction, const mesh::Mesh &inputMesh, const mesh::Mesh &outputMesh, std::vector<bool> deadAxis, Polynomial polynomial)
173 {
174  // Convert dead axis vector into an active axis array so that we can handle the reduction more easily
175  std::array<bool, 3> activeAxis({{false, false, false}});
176  std::transform(deadAxis.begin(), deadAxis.end(), activeAxis.begin(), [](const auto ax) { return !ax; });
177  // First, assemble the interpolation matrix and check the invertability
178  _qrMatrixC = buildMatrixCLU(basisFunction, inputMesh, activeAxis, polynomial).colPivHouseholderQr();
179 
180  PRECICE_CHECK(_qrMatrixC.isInvertible(),
181  "The interpolation matrix of the RBF mapping from mesh {} to mesh {} is not invertable. "
182  "This means that the mapping problem is not well-posed. "
183  "Please check if your coupling meshes are correct. Maybe you need to fix axis-aligned mapping setups "
184  "by marking perpendicular axes as dead?",
185  inputMesh.getName(), outputMesh.getName());
186 
187  // Second, assemble evaluation matrix
188  _matrixA = buildMatrixA(basisFunction, inputMesh, outputMesh, activeAxis, polynomial);
189 
190  // In case we deal with separated polynomials, we need dedicated matrices for the polynomial contribution
191  if (polynomial == Polynomial::SEPARATE) {
192 
193  // 1. Allocate memory for these matrices
194  // 4 = 1 + dimensions(3) = maximum number of polynomial parameters
195  const unsigned int polyParams = 4 - std::count(activeAxis.begin(), activeAxis.end(), false);
196  _matrixQ.resize(inputMesh.vertices().size(), polyParams);
197  _matrixV.resize(outputMesh.vertices().size(), polyParams);
198 
199  // 2. fill the matrices: Q for the inputMesh, V for the outputMesh
200  fillPolynomialEntries(_matrixQ, inputMesh, 0, activeAxis);
201  fillPolynomialEntries(_matrixV, outputMesh, 0, activeAxis);
202 
203  // 3. compute decomposition
