75 {
76  PRECICE_TRACE();
77 
78  PRECICE_DEBUG("currentColumns={}", _currentColumns);
79  if (_currentColumns > 1) {
80  PRECICE_ERROR("Truncated IMVJ is no longer supported. Please use IMVJ with restart mode instead.");
81  PRECICE_DEBUG("compute update with QR-dec");
82  //computeNewtonFactorsQRDecomposition(cplData, xUpdate);
83  } else {
84  PRECICE_DEBUG("compute update with Broyden");
85  // ------------- update inverse Jacobian -----------
86  // ------------- Broyden Update
87  //
88  // J_inv = J_inv_n + (w- J_inv_n*v)*v^T/|v|_l2
89  // ----------------------------------------- -------
90 
91  Eigen::VectorXd v = _matrixV.col(0);
92  Eigen::VectorXd w = _matrixW.col(0);
93  Eigen::MatrixXd JUpdate = Eigen::MatrixXd::Zero(_invJacobian.rows(), _invJacobian.cols());
94 
95  PRECICE_DEBUG("took latest column of V,W");
96 
97  double dotproductV = v.dot(v);
98  Eigen::VectorXd tmp = _oldInvJacobian * v; // J_inv*v
99  tmp = w - tmp; // (w-J_inv*v)
100  tmp = tmp / dotproductV; // (w-J_inv*v)/|v|_l2
101  PRECICE_DEBUG("did step (W-J_inv*v)/|v|");
102 
103  PRECICE_ASSERT(tmp.size() == v.size(), tmp.size(), v.size());
104  JUpdate = tmp * v.transpose();
105  PRECICE_DEBUG("multiplied (w-J_inv*v)/|v| * v^T");
106 
107  PRECICE_ASSERT(_invJacobian.rows() == JUpdate.rows(), _invJacobian.rows(), JUpdate.rows());
108  _invJacobian = _oldInvJacobian + JUpdate;
109 
110  // solve delta_x = - J_inv*residuals
111  xUpdate = _invJacobian * (-_residuals);
112  }
113
