435 void Participant::exportIntermediate(IntermediateExport exp)
436 {
437  for (const io::ExportContext &context : exportContexts()) {
438  if (exp.complete && (context.everyNTimeWindows > 0) && (exp.timewindow % context.everyNTimeWindows == 0)) {
439  for (const MeshContext *meshContext : usedMeshContexts()) {
440  auto &mesh = *meshContext->mesh;
441  PRECICE_DEBUG("Exporting mesh {} for timewindow {} to location \"{}\"", mesh.getName(), exp.timewindow, context.location);
442  context.exporter->doExport(fmt::format("{}-{}.dt{}", mesh.getName(), getName(), exp.timewindow), context.location, mesh);
443  }
444  }
445 
446  if (context.everyIteration) {
447  for (const MeshContext *meshContext : usedMeshContexts()) {
448  auto &mesh = *meshContext->mesh;
449  PRECICE_DEBUG("Exporting mesh {} for iteration {} to location \"{}\"", meshContext->mesh->getName(), exp.iteration, context.location);
451  context.exporter->doExport(fmt::format("{}-{}.it{}", mesh.getName(), getName(), exp.iteration), context.location, mesh);
452  }
453  }
454  }
455 
456  if (exp.complete) {
457  // Export watch point data
458  for (const PtrWatchPoint &watchPoint : watchPoints()) {
459  watchPoint->exportPointData(exp.time);
460  }
