353 void ParticipantState::exportIntermediate(IntermediateExport exp)
354 {
355  for (const io::ExportContext &context : exportContexts()) {
356  if (exp.complete && (context.everyNTimeWindows > 0) && (exp.timewindow % context.everyNTimeWindows == 0)) {
357  for (const MeshContext *meshContext : usedMeshContexts()) {
358  auto &mesh = *meshContext->mesh;
359  PRECICE_DEBUG("Exporting mesh {} for timewindow {} to location \"{}\"", mesh.getName(), exp.timewindow, context.location);
360  context.exporter->doExport(fmt::format("{}-{}.dt{}", mesh.getName(), getName(), exp.timewindow), context.location, mesh);
361  }
362  }
363 
364  if (context.everyIteration) {
365  for (const MeshContext *meshContext : usedMeshContexts()) {
366  auto &mesh = *meshContext->mesh;
367  PRECICE_DEBUG("Exporting mesh {} for iteration {} to location \"{}\"", meshContext->mesh->getName(), exp.iteration, context.location);
369  context.exporter->doExport(fmt::format("{}-{}.it{}", mesh.getName(), getName(), exp.iteration), context.location, mesh);
370  }
371  }
372  }
373 
374  if (exp.complete) {
375  // Export watch point data
376  for (const PtrWatchPoint &watchPoint : watchPoints()) {
377  watchPoint->exportPointData(exp.time);
378  }
379

421 std::string ParticipantState::hintForMeshData(std::string_view mesh, std::string_view data) const
422 {
423  PRECICE_ASSERT(hasMesh(mesh));

426

429  for (const auto &[_, mc] : _meshContexts) {
