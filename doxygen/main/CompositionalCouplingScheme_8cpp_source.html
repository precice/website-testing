85 void CompositionalCouplingScheme::advance()
86 {
87  PRECICE_TRACE();
88  bool moreSchemesToHandle = false;
89  do {
90  for (SchemesIt it = _activeSchemesBegin; it != _activeSchemesEnd; it++) {
91  if (not it->onHold) {
92  it->scheme->advance();
93  }
94  }
95  moreSchemesToHandle = determineActiveCouplingSchemes();
96  if (moreSchemesToHandle) {
97  // The new schemes to be handled in this advance also need the time that
98  // has been computed so far. This time can't be added in the solver call
99  // to addComputedTime(), since there the schemes are not active yet.
100  addComputedTime(_lastAddedTime);
101  }
102  } while (moreSchemesToHandle);
103  _lastAddedTime = 0.0;
104 }

317 bool CompositionalCouplingScheme::determineActiveCouplingSchemes()
318 {
319  PRECICE_TRACE();
320  bool newActiveSchemes = false;
321  const std::string &writeCheckpoint = constants::actionWriteIterationCheckpoint();
322  const std::string &readCheckpoint = constants::actionReadIterationCheckpoint();
323  if (_activeSchemesBegin == _activeSchemesEnd) {
324  PRECICE_DEBUG("Case After Init");
325  // First call after initialization of all coupling schemes. All coupling
326  // schemes are set active up to (but not including) the first explicit
327  // scheme after an implicit scheme.
328  _activeSchemesBegin = _couplingSchemes.begin();
329  _activeSchemesEnd = _couplingSchemes.begin();
330  advanceActiveCouplingSchemes();
331  newActiveSchemes = true;
332  } else {
333  PRECICE_DEBUG("Normal Case");
334  // Redetermine active schemes. First, all preceding explicit schemes are
335  // removed. Then, all remaining implicit schemes are checked for the
336  // convergence of iterations (this is given when an iteration checkpoint
337  // should be created). If all are converged, a next set of active schemes
338  // is determined.
339 
340  // Remove preceding explicit schemes
341  while (_activeSchemesBegin != _activeSchemesEnd) {
342  bool explicitScheme = true;
343  explicitScheme &= not _activeSchemesBegin->scheme->isActionRequired(writeCheckpoint);
344  explicitScheme &= not _activeSchemesBegin->scheme->isActionRequired(readCheckpoint);
345  if (explicitScheme) {
346  _activeSchemesBegin++;
347  PRECICE_DEBUG("Remove preceding explicit scheme");
348  } else {
349  break;
350  }
351  }
352 
353  // Check implicit schemes for convergence and remove if converged
354  bool converged = true;
355  for (SchemesIt it = _activeSchemesBegin; it != _activeSchemesEnd; it++) {
356  if (it->scheme->isActionRequired(readCheckpoint)) {
357  converged = false;
358  PRECICE_DEBUG("Non converged implicit scheme");
359  } else if (it->scheme->isActionRequired(writeCheckpoint) || not it->scheme->isCouplingOngoing()) {
360  it->onHold = true;
361  PRECICE_DEBUG("Put converged/finished implicit scheme on hold");
362  }
363  }
364  if (converged) {
365  PRECICE_DEBUG("Active implicit schemes converged");
366  for (SchemesIt it = _activeSchemesBegin; it != _activeSchemesEnd; it++) {
367  it->onHold = false;
368  }
369  _activeSchemesBegin = _activeSchemesEnd;
370  }
371 
372  // Determine next set of active schemes if current is empty
373  if (_activeSchemesBegin == _activeSchemesEnd) {
374  if (_activeSchemesBegin == _couplingSchemes.end()) {
375  PRECICE_DEBUG("Through with all coupling schemes");
376  // All coupling schemes are through
377  _activeSchemesBegin = _couplingSchemes.begin();
378  _activeSchemesEnd = _couplingSchemes.begin();
379  advanceActiveCouplingSchemes();
380  // newActiveSchemes stays false, since the current it/dt is complete
381  } else {
382  PRECICE_DEBUG("Coupling schemes remaining");
383  advanceActiveCouplingSchemes();
384  newActiveSchemes = true;
385  }
386  }
387  }
388  PRECICE_DEBUG("return newActiveSchemes={}", newActiveSchemes);
389  return newActiveSchemes;
390 }

392 void CompositionalCouplingScheme::advanceActiveCouplingSchemes()
393 {
394  PRECICE_TRACE();
395  const std::string &writeCheckpoint = constants::actionWriteIterationCheckpoint();
396  bool iterating = false;
397  while (_activeSchemesEnd != _couplingSchemes.end()) {
398  if (_activeSchemesEnd->scheme->isActionRequired(writeCheckpoint)) {
399  PRECICE_DEBUG("Found implicit scheme");
400  iterating = true;
401  }
402  if (iterating && (not _activeSchemesEnd->scheme->isActionRequired(writeCheckpoint))) {
403  PRECICE_DEBUG("Found explicit scheme after implicit scheme");
404  break;
405  }
406  _activeSchemesEnd++;
407  PRECICE_DEBUG("Advanced active schemes by one new scheme");
408  }
409  PRECICE_ASSERT(_activeSchemesBegin != _activeSchemesEnd);
410 }
