96 {

99  if (receivesInitializedData()) {

117 bool SerialCouplingScheme::exchangeDataAndAccelerate()
118 {
119  bool convergence = true;
120 
121  if (doesFirstStep()) { // first participant
122  PRECICE_DEBUG("Sending data...");
123  sendTimeWindowSize();
124  sendData(getM2N(), getSendData());

128  PRECICE_DEBUG("Receiving data...");

131  } else { // second participant

134  convergence = doImplicitStep();
135  sendConvergence(getM2N(), convergence);
136  }
137  PRECICE_DEBUG("Sending data...");
138  sendData(getM2N(), getSendData());
139  // the second participant does not want new data in the last iteration of the last time window
140  if (isCouplingOngoing() || (isImplicitCouplingScheme() && not convergence)) {
141  receiveAndSetTimeWindowSize();
142  PRECICE_DEBUG("Receiving data...");
143  receiveData(getM2N(), getReceiveData());
144  checkDataHasBeenReceived();
145  }
146  }
