169

171  "Data with name \"{0}\" associated to mesh \"{1}\" not found on configuration of acceleration. "
172  "Add \"{0}\" to the \"<mesh name={1}>\" tag, or change the data name in the acceleration scheme.",
173  dataName, _meshName);

177  _config.dataIDs.push_back(data->getID());

179 
180  _neededMeshes.push_back(_meshName);
181  } else if (callingTag.getName() == TAG_INIT_RELAX) {
182  _config.relaxationFactor = callingTag.getDoubleAttributeValue(ATTR_VALUE);
183  if (callingTag.hasAttribute(ATTR_ENFORCE)) {
184  _config.forceInitialRelaxation = callingTag.getBooleanAttributeValue(ATTR_ENFORCE);
185  } else {
186  _config.forceInitialRelaxation = false;
187  }
188  } else if (callingTag.getName() == TAG_MAX_USED_ITERATIONS) {
189  _config.maxIterationsUsed = callingTag.getIntAttributeValue(ATTR_VALUE);
190  } else if (callingTag.getName() == TAG_TIME_WINDOWS_REUSED) {
191  _config.timeWindowsReused = callingTag.getIntAttributeValue(ATTR_VALUE);
192  } else if (callingTag.getName() == TAG_FILTER) {
193  const auto &f = callingTag.getStringAttributeValue(ATTR_TYPE);
194  if (f == VALUE_QR1FILTER) {
195  _config.filter = Acceleration::QR1FILTER;
196  } else if (f == VALUE_QR1_ABSFILTER) {
197  _config.filter = Acceleration::QR1FILTER_ABS;
198  } else if (f == VALUE_QR2FILTER) {
199  _config.filter = Acceleration::QR2FILTER;
200  } else {
201  PRECICE_ASSERT(false);
202  }
203  _config.singularityLimit = callingTag.getDoubleAttributeValue(ATTR_SINGULARITYLIMIT);
204  } else if (callingTag.getName() == TAG_PRECONDITIONER) {
205  _config.preconditionerType = callingTag.getStringAttributeValue(ATTR_TYPE);
206  _config.precond_nbNonConstTWindows = callingTag.getIntAttributeValue(ATTR_PRECOND_NONCONST_TIME_WINDOWS);
207  } else if (callingTag.getName() == TAG_IMVJRESTART) {
208 
209  if (_config.alwaysBuildJacobian)
210  PRECICE_ERROR("IMVJ cannot be in restart mode while parameter always-build-jacobian is set to true. "
211  "Please remove 'always-build-jacobian' from the configuration file or do not run in restart mode.");
212 
213 #ifndef PRECICE_NO_MPI

215  const auto &f = callingTag.getStringAttributeValue(ATTR_TYPE);
216  if (f == VALUE_NO_RESTART) {
217  _config.imvjRestartType = MVQNAcceleration::NO_RESTART;
218  } else if (f == VALUE_ZERO_RESTART) {
219  _config.imvjRestartType = MVQNAcceleration::RS_ZERO;
220  } else if (f == VALUE_LS_RESTART) {
221  _config.imvjRSLS_reusedTimeWindows = callingTag.getIntAttributeValue(ATTR_RSLS_REUSED_TIME_WINDOWS);
222  _config.imvjRestartType = MVQNAcceleration::RS_LS;
223  } else if (f == VALUE_SVD_RESTART) {
224  _config.imvjRSSVD_truncationEps = callingTag.getDoubleAttributeValue(ATTR_RSSVD_TRUNCATIONEPS);
225  _config.imvjRestartType = MVQNAcceleration::RS_SVD;
226  } else if (f == VALUE_SLIDE_RESTART) {
227  _config.imvjRestartType = MVQNAcceleration::RS_SLIDE;
228  } else {
229  _config.imvjChunkSize = 0;
230  PRECICE_ASSERT(false);
231  }

234 #endif

238 void AccelerationConfiguration::xmlEndTagCallback(
239  const xml::ConfigurationContext &context,
240  xml::XMLTag & callingTag)
241 {
242  PRECICE_TRACE(callingTag.getName());
243  if (callingTag.getNamespace() == TAG) {
244 
245  //create preconditioner
246  if (callingTag.getName() == VALUE_IQNILS || callingTag.getName() == VALUE_MVQN) {
247 
248  // if imvj restart-mode is of type RS-SVD, max number of non-const preconditioned time windows is limited by the chunksize
249  if (callingTag.getName() == VALUE_MVQN && _config.imvjRestartType > 0)
250  if (_config.precond_nbNonConstTWindows > _config.imvjChunkSize)
251  _config.precond_nbNonConstTWindows = _config.imvjChunkSize;
252 
253  if (_config.preconditionerType == VALUE_CONSTANT_PRECONDITIONER) {
254  std::vector<double> factors;
255  for (int id : _config.dataIDs) {
256  factors.push_back(_config.scalings[id]);
257  }
258  _preconditioner = PtrPreconditioner(new ConstantPreconditioner(factors));
259  } else if (_config.preconditionerType == VALUE_VALUE_PRECONDITIONER) {
260  _preconditioner = PtrPreconditioner(new ValuePreconditioner(_config.precond_nbNonConstTWindows));
261  } else if (_config.preconditionerType == VALUE_RESIDUAL_PRECONDITIONER) {
262  _preconditioner = PtrPreconditioner(new ResidualPreconditioner(_config.precond_nbNonConstTWindows));
263  } else if (_config.preconditionerType == VALUE_RESIDUAL_SUM_PRECONDITIONER) {
264  _preconditioner = PtrPreconditioner(new ResidualSumPreconditioner(_config.precond_nbNonConstTWindows));
265  } else {
266  // no preconditioner defined
267  std::vector<double> factors;
268  for (int id = 0; id < static_cast<int>(_config.dataIDs.size()); ++id) {
269  factors.push_back(1.0);
270  }
271  _preconditioner = PtrPreconditioner(new ConstantPreconditioner(factors));
272  }
273  }
274 
275  if (callingTag.getName() == VALUE_CONSTANT) {
276  _acceleration = PtrAcceleration(
277  new ConstantRelaxationAcceleration(
278  _config.relaxationFactor, _config.dataIDs));
279  } else if (callingTag.getName() == VALUE_AITKEN) {
280  _acceleration = PtrAcceleration(
281  new AitkenAcceleration(
282  _config.relaxationFactor, _config.dataIDs));
283  } else if (callingTag.getName() == VALUE_IQNILS) {
284  _acceleration = PtrAcceleration(
285  new IQNILSAcceleration(
286  _config.relaxationFactor,
287  _config.forceInitialRelaxation,
288  _config.maxIterationsUsed,
289  _config.timeWindowsReused,
290  _config.filter, _config.singularityLimit,
291  _config.dataIDs,
292  _preconditioner));
293  } else if (callingTag.getName() == VALUE_MVQN) {
294 #ifndef PRECICE_NO_MPI
295  _acceleration = PtrAcceleration(
296  new MVQNAcceleration(
297  _config.relaxationFactor,
298  _config.forceInitialRelaxation,
299  _config.maxIterationsUsed,
300  _config.timeWindowsReused,
301  _config.filter, _config.singularityLimit,
302  _config.dataIDs,
303  _preconditioner,
304  _config.alwaysBuildJacobian,
305  _config.imvjRestartType,
306  _config.imvjChunkSize,
307  _config.imvjRSLS_reusedTimeWindows,
308  _config.imvjRSSVD_truncationEps));
309 #else
310  PRECICE_ERROR("Acceleration IQN-IMVJ only works if preCICE is compiled with MPI");
311 #endif
312  } else if (callingTag.getName() == VALUE_BROYDEN) {
313  _acceleration = PtrAcceleration(
314  new BroydenAcceleration(
315  _config.relaxationFactor,
316  _config.forceInitialRelaxation,
317  _config.maxIterationsUsed,
318  _config.timeWindowsReused,
319  _config.filter, _config.singularityLimit,
320  _config.dataIDs,
321  _preconditioner));
322  } else {
323  PRECICE_ASSERT(false);
324  }
325  }
