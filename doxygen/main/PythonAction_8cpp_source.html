28 std::string python_error_as_string()
29 {
30  PyObject *ptype, *pvalue, *ptraceback;
31  PyErr_Fetch(&ptype, &pvalue, &ptraceback);
32  if (ptype == nullptr) {
33  return "<no error available>";
34  } else {
35  // pvalue and ptraceback may be NULL
36  // We don't need the type or the traceback, so we dereference them straight away
37  Py_DECREF(ptype);
38  Py_XDECREF(ptraceback); // may be NULL
39 
40  if (pvalue == nullptr) {
41  return "<no error message available>";
42  }
43  wchar_t *wmessage = PyUnicode_AsWideCharString(pvalue, nullptr);
44  Py_DECREF(pvalue);
45 
46  if (wmessage) {
47  auto message = utils::truncate_wstring_to_string(wmessage);
48  PyMem_Free(wmessage);
49  return message;
50  } else {
51  return "<fetching error message failed>";
52  }
53  }

68 std::vector<std::string> python_func_args(PyObject *const func)
69 {
70  PyObject *const getfullargspec_function = getfullargspec();
71 
72  // Call the inspect.getfullargspec function.
73  PyObject *const argspec_call_args = PyTuple_New(1);
74  PyTuple_SetItem(argspec_call_args, 0, func);
75  PyObject *const argspec = PyObject_CallObject(getfullargspec_function, argspec_call_args);
76  Py_DECREF(argspec_call_args);
77  Py_DECREF(getfullargspec_function);
78 
79  // Get args from argspec.

81  Py_ssize_t const num_args = PyList_Size(f_args);
82  std::vector<std::string> arg_names;
83  for (Py_ssize_t i = 0; i < num_args; ++i) {
84  PyObject *const arg = PyList_GetItem(f_args, i);
85  PyObject *const arg_repr = PyObject_Repr(arg);
86  PyObject *const arg_str = PyUnicode_AsASCIIString(arg_repr);
87  arg_names.emplace_back(PyBytes_AS_STRING(arg_str));
88  Py_DECREF(arg);
89  Py_DECREF(arg_repr);
90  Py_DECREF(arg_str);
