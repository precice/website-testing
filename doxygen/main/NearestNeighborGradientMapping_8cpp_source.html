39 {
40 
41  // Initialize the offsets list
42  _offsetsMatched.resize(_vertexIndices.size());
43 
44  // Calculate offsets
45  for (size_t i = 0; i < _vertexIndices.size(); ++i) {
46 
47  const auto &matchedVertexCoords = searchSpace.get()->vertices()[_vertexIndices[i]].getCoords();
48  const auto &sourceVertexCoords = origins->vertices()[i].getCoords();
49 
50  // We calculate the distances uniformly for consistent mapping constraint as the difference (output - input)
51  // For consistent mapping: the source is the output vertex and the matched vertex is the input since we iterate over all outputs
52  // and assign each exactly one vertex form the search space, which are our origins vertices.
53  _offsetsMatched[i] = sourceVertexCoords - matchedVertexCoords;
54  }
