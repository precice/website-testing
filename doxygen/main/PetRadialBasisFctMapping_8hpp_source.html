604  auto a = petsc::Vector::allocate(_matrixQ, "a", petsc::Vector::RIGHT); // holds the solution of the LS polynomial

609  for (int dim = 0; dim < valueDim; dim++) {
610  printMappingInfo(inputDataID, dim);

613  std::vector<PetscScalar> inVals;
614  inVals.reserve(this->input()->vertices().size());
615  std::vector<PetscInt> inIdx;
616  inIdx.reserve(this->input()->vertices().size());
617  for (size_t i = 0; i < this->input()->vertices().size(); ++i) {
618  if (not this->input()->vertices()[i].isOwner())
619  continue;
620  inVals.emplace_back(inValues[i * valueDim + dim]);
621  inIdx.emplace_back(inIdx.size() + in.ownerRange().first + localPolyparams);
622  }

624  CHKERRV(ierr);
625  in.assemble();
626 
627  if (_polynomial == Polynomial::SEPARATE) {
628  switch (_QRsolver.solve(in, a)) {
629  case (petsc::KSPSolver::SolverResult::Converged):
630  PRECICE_DEBUG("The polynomial QR system of the RBF mapping from mesh {} to mesh {} converged. {}",
631  this->input()->getName(), this->output()->getName(), _QRsolver.summaryFor(in));
632  break;
633  case (petsc::KSPSolver::SolverResult::Stopped):
634  PRECICE_WARN("The polynomial QR system of the RBF mapping from mesh {} to mesh {} has not converged. "
635  "This means most probably that the mapping problem is not well-posed or your relative tolerance is too conservative. "
636  "Please check if your coupling meshes are correct. "
637  "Maybe you need to fix axis-aligned mapping setups by marking perpendicular axes as dead? {}",
638  this->input()->getName(), this->output()->getName(), _QRsolver.summaryFor(in));
639  break;
640  case (petsc::KSPSolver::SolverResult::Diverged):
641  KSPView(_QRsolver, PETSC_VIEWER_STDOUT_WORLD);
642  PRECICE_ERROR("The polynomial QR system of the RBF mapping from mesh {} to mesh {} has diverged. "
643  "This means most probably that the mapping problem is not well-posed. "
644  "Please check if your coupling meshes are correct. "
645  "Maybe you need to fix axis-aligned mapping setups by marking perpendicular axes as dead? {}",
646  this->input()->getName(), this->output()->getName(), _QRsolver.summaryFor(in));
647  break;
648  }
649 
650  VecScale(a, -1);
651  // in = Q * a + in
652  MatMultAdd(_matrixQ, a, in, in); // Subtract the polynomial from the input values
653  }
654 
655  petsc::Vector &p = std::get<0>( // Save and reuse the solution from the previous iteration
656  previousSolution.emplace(std::piecewise_construct,
657  std::forward_as_tuple(inputDataID + outputDataID * 10 + dim * 100),
658  std::forward_as_tuple(petsc::Vector::allocate(_matrixC, "p"))))
659  ->second;
660 
661  utils::Event eSolve("map.pet.solveConsistent.From" + this->input()->getName() + "To" + this->output()->getName(), precice::syncMode);
662  const auto solverResult = _solver.solve(in, p);
663  eSolve.addData("Iterations", _solver.getIterationNumber());
664  eSolve.stop();
665 
666  switch (solverResult) {
667  case (petsc::KSPSolver::SolverResult::Converged):
668  PRECICE_DEBUG("The linear system of the RBF mapping from mesh {} to mesh {} converged. {}",
669  this->input()->getName(), this->output()->getName(), _solver.summaryFor(in));
670  break;
671  case (petsc::KSPSolver::SolverResult::Stopped):
672  PRECICE_WARN("The linear system of the RBF mapping from mesh {} to mesh {} has not converged. "
673  "This means most probably that the mapping problem is not well-posed or your relative tolerance is too conservative. "
674  "Please check if your coupling meshes are correct. "
675  "Maybe you need to fix axis-aligned mapping setups by marking perpendicular axes as dead? {}",
676  this->input()->getName(), this->output()->getName(), _solver.summaryFor(in));
677  break;
678  case (petsc::KSPSolver::SolverResult::Diverged):
679  KSPView(_solver, PETSC_VIEWER_STDOUT_WORLD);
680  PRECICE_ERROR("The linear system of the RBF mapping from mesh {} to mesh {} has diverged. "
681  "This means most probably that the mapping problem is not well-posed. "
682  "Please check if your coupling meshes are correct. "
683  "Maybe you need to fix axis-aligned mapping setups by marking perpendicular axes as dead? {}",
684  this->input()->getName(), this->output()->getName(), _solver.summaryFor(in));
685  break;
686  }
687 
688  ierr = MatMult(_matrixA, p, out);
689  CHKERRV(ierr);
690 
691  if (useRescaling and _polynomial == Polynomial::SEPARATE) {
692  ierr = VecPointwiseDivide(out, out, oneInterpolant);
693  CHKERRV(ierr);
694  }
695 
696  if (_polynomial == Polynomial::SEPARATE) {
697  // scale it back to add the polynomial
698  ierr = VecScale(a, -1);
699  // out = V * a + out
700  ierr = MatMultAdd(_matrixV, a, out, out);
701  CHKERRV(ierr);
702  }
703  VecChop(out, 1e-9);
704 
705  // Copy mapped data to output data values
706  ierr = VecGetArrayRead(out, &vecArray);
707  CHKERRV(ierr);
708  int size = out.getLocalSize();
709  for (int i = 0; i < size; i++) {
710  outValues[i * valueDim + dim] = vecArray[i];
711  }
712 
713  VecRestoreArrayRead(out, &vecArray);
714  }

805  case (petsc::KSPSolver::SolverResult::Converged):
806  PRECICE_DEBUG("The linear system of the RBF mapping from mesh {} to mesh {} converged. {}",
807  this->input()->getName(), this->output()->getName(), _solver.summaryFor(au));
808  break;
809  case (petsc::KSPSolver::SolverResult::Stopped):
810  PRECICE_WARN("The linear system of the RBF mapping from mesh {} to mesh {} has not converged. "
811  "This means most probably that the mapping problem is not well-posed or your relative tolerance is too conservative. "
812  "Please check if your coupling meshes are correct. "
813  "Maybe you need to fix axis-aligned mapping setups by marking perpendicular axes as dead? {}",
814  this->input()->getName(), this->output()->getName(), _solver.summaryFor(au));
815  break;
816  case (petsc::KSPSolver::SolverResult::Diverged):
817  KSPView(_solver, PETSC_VIEWER_STDOUT_WORLD);
818  PRECICE_ERROR("The linear system of the RBF mapping from mesh {} to mesh {} "

821  "Maybe you need to fix axis-aligned mapping setups by marking perpendicular axes as dead? {}",
822  this->input()->getName(), this->output()->getName(), _solver.summaryFor(au));
823  break;
824  }
825  }
826 
827  VecChop(out, 1e-9);
828 
829  // Copy mapped data to output data values
830  const PetscScalar *outArray;
