263 void Parallel::splitCommunicator(const std::string &groupName)
264 {
265 #ifndef PRECICE_NO_MPI
266  PRECICE_TRACE(groupName);
267 
268  // Exchange group information
269  PRECICE_DEBUG("Exchange group information");
270  //_accessorGroups.clear(); // Makes reinitialization possible
271 
272  CommStatePtr baseState = current();
273  MPI_Comm globalComm = baseState->comm;
274  int rank = baseState->rank();
275  int size = baseState->size();
276 
277  PRECICE_ASSERT(size > 1, "Splitting a communicator or size 1 is not possible!");
278 
279  std::vector<AccessorGroup> accessorGroups;
280  com::MPIDirectCommunication com;
281 
282  if (rank == 0) { // Primary rank
283  // Step 1 receive groupNames from Salves and setup accessorGroups
284  std::map<std::string, int> groupMap; // map from names to group ID
285  groupMap[groupName] = 0;
286  AccessorGroup newGroup;
287  newGroup.id = 0;
288  newGroup.size = 1;
289  newGroup.leaderRank = 0;
290  newGroup.name = groupName;
291  accessorGroups.push_back(newGroup);
292  for (int i = 1; i < size; i++) {
293  std::string name;
294  com.receive(name, i); // Receive group name from all ranks
295  if (groupMap.find(name) == groupMap.end()) {
296  groupMap[name] = accessorGroups.size();
297  AccessorGroup newGroup;
298  newGroup.id = accessorGroups.size();
299  newGroup.size = 1;
300  newGroup.leaderRank = i;
301  newGroup.name = name;
302  accessorGroups.push_back(newGroup);
303  } else {
304  accessorGroups[groupMap[name]].size++;
305  }
306  }
307  // Step 2 send groupCount to Primary rank
308  auto groupCount = static_cast<int>(accessorGroups.size());
309  MPI_Bcast(&groupCount, 1, MPI_INT, 0, globalComm);
310  PRECICE_ASSERT(groupCount > 1, "Calling split with a single group is not permitted!");
311 
312  // Step 3 send AccessorGroups to SecondaryRanks
313  for (const AccessorGroup &group : accessorGroups) {
314  // @TODO can we use broadcast as the primary rank sends this to everyone else?
315  for (int i = 1; i < size; i++) {
316  com.send(group.name, i);
317  com.send(group.leaderRank, i);
318  com.send(group.id, i);
319  com.send(group.size, i);
320  }
321  }
322  } else { // rank != 0
323  // Step 1 send groupName to Primary rank
324  com.send(groupName, 0);
325  // Step 2 receive groupCount from Primary rank
326  int groupCount = -1;
327  MPI_Bcast(&groupCount, 1, MPI_INT, 0, globalComm);
328  PRECICE_ASSERT(groupCount > 1, "Calling split with a single group is not permitted!");
329 
330  // Step 3 receive AccessorGroups from Primary rank
331  for (int i = 0; i < groupCount; i++) {
332  AccessorGroup newGroup;
333  com.receive(newGroup.name, 0);
334  com.receive(newGroup.leaderRank, 0);
335  com.receive(newGroup.id, 0);
336  com.receive(newGroup.size, 0);
337  accessorGroups.emplace_back(std::move(newGroup));
338  }
339  }
340 
341  // Step 4 split into groups
342  PRECICE_DEBUG("Split groups");
343  auto thisGroup = std::find_if(accessorGroups.begin(), accessorGroups.end(), [groupName](const AccessorGroup &group) { return group.name == groupName; });
344  PRECICE_ASSERT(thisGroup != accessorGroups.end(), "This requested groupName is not in accessorGroups!", groupName);
345 
346  CommStatePtr newState;
347  const bool restrictToSelf = std::all_of(accessorGroups.begin(), accessorGroups.end(), [](const AccessorGroup &group) { return group.size == 1; });
348  if (restrictToSelf) {
349  PRECICE_DEBUG("Split to Comm Self");
350  newState = CommState::self();
351  } else {
352  PRECICE_DEBUG("Split to new Communicator");
353  // Create a new communicator that contains only ranks of my group
354  MPI_Comm newComm = MPI_COMM_NULL;
355  MPI_Comm_split(globalComm, thisGroup->id, rank, &newComm);
356  // Assemble and set new state
357  newState = CommState::fromComm(newComm);
358  }
359 
360 #ifndef NDEBUG
361  PRECICE_DEBUG("Detected {} groups", accessorGroups.size());
362  for (const AccessorGroup &group : accessorGroups) {
363  PRECICE_DEBUG("Group {}: name = {}, leaderRank = {}, size = {}",
364  group.id, group.name, group.leaderRank, group.size);
365  }
366 #endif // NDEBUG
367 
368  newState->groups = std::move(accessorGroups);
369  pushState(newState);
370 
371 #endif // not PRECICE_NO_MPI
372 }

479 void Parallel::restrictCommunicator(int newSize)
480 {
481  PRECICE_TRACE(newSize);
482  PRECICE_ASSERT(newSize > 0, "Cannot restrict a communicator to nothing!");
483 
484 #ifndef PRECICE_NO_MPI
485  auto baseState = current();
486  const auto size = baseState->size();
487  const auto rank = baseState->rank();
488 
489  PRECICE_ASSERT(newSize <= size, "Requested more ranks than the Communicator can provide!");
490 
491  // A single rank can use MPI_COMM_SELF, nothing else required
492  if (newSize == 1) {
493  // @todo verify if the barrier is required
494  // PRECICE_DEBUG("Barrier");
495  // MPI_Barrier(_globalCommunicator);
496  if (rank == 0) {
497  PRECICE_DEBUG("Restricted to COMM_SELF");
498  pushState(CommState::self());
499  return;
500  } else {
501  PRECICE_DEBUG("This rank remains unused after the restriction.");
502  pushState(CommState::null());
503  return;
504  }
505  }
506 
507  // If the requested size is the same as the capacity, then simply duplicate the comm
508  if (newSize == size) {
509  PRECICE_DEBUG("Restriction to capacity: duplicating Comm");
510  MPI_Comm copiedComm;
511  MPI_Comm_dup(baseState->comm, &copiedComm);
512  pushState(CommState::fromComm(copiedComm));
513  return;
514  }
515 
516  // Create group, containing all processes of communicator
517  MPI_Group currentGroup;
518  MPI_Comm_group(baseState->comm, &currentGroup);
519 
520  // Prepare a ranks vector with 0,1,2,...,newSize-1
521  std::vector<int> ranks(newSize);
522  std::iota(ranks.begin(), ranks.end(), 0);
523 
524  // Create subgroup, containing processes contained in ranks
525  PRECICE_DEBUG("Create restricted group");
526  MPI_Group restrictedGroup;
527  MPI_Group_incl(currentGroup, ranks.size(), ranks.data(), &restrictedGroup);
528 
529  // @todo check if it has to go back down to the other group free
530  MPI_Group_free(&currentGroup);
531 
532  {
533  int restrictedGroupSize = 0;
534  MPI_Group_size(restrictedGroup, &restrictedGroupSize);
535  PRECICE_ASSERT(restrictedGroupSize == newSize, "Group size differs: restriction failed!");
536  }
537 
538  // Create communicator, containing process of restrictedGroup
539  PRECICE_DEBUG("Create restricted comm using group");
540 
541  // MPI_Comm restrictedCommunicator;
542  MPI_Comm restrictedCommunicator = MPI_COMM_NULL;
543  MPI_Comm_create(baseState->comm, restrictedGroup, &restrictedCommunicator);
544 
545  // @todo check if the we need a barrier here
546  // PRECICE_DEBUG("Barrier");
547  // MPI_Barrier(_globalCommunicator);
548  PRECICE_DEBUG("Free restricted group");
549  MPI_Group_free(&restrictedGroup);
550 
551  pushState(CommState::fromComm(restrictedCommunicator));
552 #endif // not PRECICE_NO_MPI
553 }
