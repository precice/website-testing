508 void ReceivedPartition::createOwnerInformation()
509 {
510  PRECICE_TRACE();
511  Event e("partition.createOwnerInformation." + _mesh->getName(), precice::syncMode);
512 
513  /*
514  We follow different approaches for two-level and one-level methods. For 1LI, a centeralized
515  approach is followed, while the 2LI employs a local/parallel scheme for assigning vertices
516  to corresponding ranks.
517  */
518 
519  if (m2n().usesTwoLevelInitialization()) {
520  /*
521  This function ensures that each vertex is owned by only a single rank and
522  is not shared among ranks. Initially, the vertices are checked against the
523  bounding box of each rank. If a vertex fits into only a single bounding box,
524  the vertex is assigned to that rank. If it fits to various bbs, the rank with the
525  lowest number of vertices gets ownership to keep the load as balanced as
526  possible.
527 
528  Following steps are taken:
529 
530  1- receive local bb map from primary rank
531  2- filter bb map to keep the connected ranks
532  3- own the vertices that only fit into this rank's bb
533  4- send number of owned vertices and the list of shared vertices to neighbors
534  5- for the remaining vertices: check if we have less vertices -> own it!
535  */
536 
537  // #1: receive local bb map from primary rank
538  // Define and initialize localBBMap to save local bbs
539 
540  mesh::Mesh::BoundingBoxMap localBBMap;
541  for (Rank rank = 0; rank < utils::IntraComm::getSize(); rank++) {
542  localBBMap.emplace(rank, mesh::BoundingBox(_dimensions));
543  }
544 
545  // Define a bb map to save the local connected ranks and respective boundingboxes
546  mesh::Mesh::BoundingBoxMap localConnectedBBMap;
547 
548  // store global IDs and list of possible shared vertices
549 
550  // Global IDs to be saved in:
551  std::vector<int> sharedVerticesGlobalIDs;
552  // Local IDs to be saved in:
553  std::vector<int> sharedVerticesLocalIDs;
554 
555  // store possible shared vertices in a map to communicate with neighbors, map: rank -> vertex_global_id
556  std::map<int, std::vector<int>> sharedVerticesSendMap;
557 
558  // receive list of possible shared vertices from neighboring ranks
559  std::map<int, std::vector<int>> sharedVerticesReceiveMap;
560 
561  if (utils::IntraComm::isPrimary()) {
562 
563  // Insert bounding box of primary ranks
564  localBBMap.at(0) = _bb;
565
