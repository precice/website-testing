99 Eigen::VectorXd Extrapolation::extrapolate()
100 {
101  const int usedOrder = computeUsedOrder(_extrapolationOrder, _numberOfStoredSamples);
102  PRECICE_ASSERT(_storageIsInitialized);
103 
104  if (usedOrder == 0) {
105  PRECICE_ASSERT(_numberOfStoredSamples > 0);
106  return _timeWindowsStorage.col(0);
107  }
108  Eigen::VectorXd extrapolatedValue;
109  if (usedOrder == 1) { //timesteps is increased before extrapolate is called
110  PRECICE_DEBUG("Performing first order extrapolation");
111  PRECICE_ASSERT(_numberOfStoredSamples > 1);
112  extrapolatedValue = _timeWindowsStorage.col(0) * 2.0; // = 2*x^t
113  extrapolatedValue -= _timeWindowsStorage.col(1); // = 2*x^t - x^(t-1)
114  return extrapolatedValue;
115  }
116  PRECICE_ASSERT(usedOrder == 2);
117  // uses formula given in https://doi.org/10.1016/j.compstruc.2008.11.013, p.796, Algorithm line 1
118  PRECICE_DEBUG("Performing second order extrapolation");
119  PRECICE_ASSERT(_numberOfStoredSamples > 2);
120  extrapolatedValue = _timeWindowsStorage.col(0) * 2.5; // = 2.5*x^t
121  extrapolatedValue -= _timeWindowsStorage.col(1) * 2.0; // = 2.5*x^t - 2*x^(t-1)
122  extrapolatedValue += _timeWindowsStorage.col(2) * 0.5; // = 2.5*x^t - 2*x^(t-1) + 0.5*x^(t-2)
123  return extrapolatedValue;
124 }
