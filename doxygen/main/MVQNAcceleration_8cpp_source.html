302  } else {

445

447 
448  // pending deletion: delete Wtil
449  if (_firstIteration && _timeWindowsReused == 0 && not _forceInitialRelaxation) {
450  _Wtil.conservativeResize(0, 0);
451  _resetLS = true;

500  // ------------ RESTART SVD ------------
501  if (_imvjRestartType == MVQNAcceleration::RS_SVD) {
502 
503  // we need to compute the updated SVD of the scaled Jacobian matrix
504  // |= APPLY PRECONDITIONING J_prev = Wtil^q, Z^q ===|
505  for (int i = 0; i < static_cast<int>(_WtilChunk.size()); i++) {
506  _preconditioner->apply(_WtilChunk[i]);
507  _preconditioner->revert(_pseudoInverseChunk[i], true);
508  }
509  // |=================== ===|
510 
511  Rank rankBefore = _svdJ.isSVDinitialized() ? _svdJ.rank() : 0;
512 
513  // if it is the first time window, there is no initial SVD, so take all Wtil, Z matrices
514  // otherwise, the first element of each container holds the decomposition of the current
515  // truncated SVD, i.e., Wtil^0 = \phi, Z^0 = S\psi^T, this should not be added to the SVD.
516  int q = _svdJ.isSVDinitialized() ? 1 : 0;
517

519  for (; q < static_cast<int>(_WtilChunk.size()); q++) {
520  // update SVD, i.e., PSI * SIGMA * PHI^T <-- PSI * SIGMA * PHI^T + Wtil^q * Z^q

532  auto &sigma = _svdJ.singularValues();
533  auto &phi = _svdJ.matrixPhi();
534

536  // sigma is stored local on each proc, thus, the multiplication is fully local, no communication.
537  // Z = sigma * phi^T
538  Eigen::MatrixXd Z(phi.cols(), phi.rows());
539  for (int i = 0; i < static_cast<int>(Z.rows()); i++)
540  for (int j = 0; j < static_cast<int>(Z.cols()); j++)

550

561  << '\n';
562  }

568  _pseudoInverseChunk.clear();

574  }
575

577  // V needs to be sclaed to compute the pseudo inverse
578  // W only needs to be scaled, as the design requires to store scaled
579  // matrices Wtil^0 and Z^0 as initial guess after the restart

581  _preconditioner->apply(_matrixW_RSLS);

637  if (utils::IntraComm::isPrimary() || !utils::IntraComm::isParallel()) {
638  _infostringstream << " - MVJ-RESTART" << _nbRestarts << ", mode= LS -\n used cols: " << _matrixV_RSLS.cols() << "\n R_RS: " << _RSLSreusedTimeWindows << "\n"
639  << '\n';
640  }
641 
642  // ------------ RESTART ZERO ------------
643  } else if (_imvjRestartType == MVQNAcceleration::RS_ZERO) {
644  // drop all stored Wtil^q, Z^q matrices
645  _WtilChunk.clear();
646  _pseudoInverseChunk.clear();
647 
648  PRECICE_DEBUG("MVJ-RESTART, mode=Zero");
649 
650  } else if (_imvjRestartType == MVQNAcceleration::RS_SLIDE) {
651 
652  // re-compute Wtil -- compensate for dropping of Wtil_0 and Z_0:
653  // Wtil_q <-- Wtil_q + Wtil^0 * (Z^0*V_q)
654  for (int i = static_cast<int>(_WtilChunk.size()) - 1; i >= 1; i--) {
655 
656  int colsLSSystemBackThen = _pseudoInverseChunk.front().rows();
657  PRECICE_ASSERT(colsLSSystemBackThen == _WtilChunk.front().cols(), colsLSSystemBackThen, _WtilChunk.front().cols());
658  Eigen::MatrixXd ZV = Eigen::MatrixXd::Zero(colsLSSystemBackThen, _qrV.cols());
659  // multiply: ZV := Z^q * V of size (m x m) with m=#cols, stored on each proc.
660  _parMatrixOps->multiply(_pseudoInverseChunk.front(), _matrixV, ZV, colsLSSystemBackThen, getLSSystemRows(), _qrV.cols());
661  // multiply: Wtil^0 * (Z_0*V) dimensions: (n x m) * (m x m), fully local and embarrassingly parallel
662  Eigen::MatrixXd tmp = Eigen::MatrixXd::Zero(_qrV.rows(), _qrV.cols());
663  tmp = _WtilChunk.front() * ZV;
664  _WtilChunk[i] += tmp;
665 
666  // drop oldest pair Wtil_0 and Z_0
667  PRECICE_ASSERT(not _WtilChunk.empty());
668  PRECICE_ASSERT(not _pseudoInverseChunk.empty())
669  _WtilChunk.erase(_WtilChunk.begin());
670  _pseudoInverseChunk.erase(_pseudoInverseChunk.begin());
671  }
672 
673  } else if (_imvjRestartType == MVQNAcceleration::NO_RESTART) {
674  PRECICE_ASSERT(false); // should not happen, in this case _imvjRestart=false
675  } else {
676  PRECICE_ASSERT(false);
677  }
678 }

681 void MVQNAcceleration::specializedIterationsConverged(
682  const DataMap &cplData)
683 {
684  PRECICE_TRACE();
685 
686  // truncate V_RSLS and W_RSLS matrices according to _RSLSreusedTimeWindows
687  if (_imvjRestartType == RS_LS) {
688  if (_matrixCols_RSLS.front() == 0) { // Did only one iteration
689  _matrixCols_RSLS.pop_front();
690  }
691  if (_RSLSreusedTimeWindows == 0) {
692  _matrixV_RSLS.resize(0, 0);
693  _matrixW_RSLS.resize(0, 0);
694  _matrixCols_RSLS.clear();
695  } else if (static_cast<int>(_matrixCols_RSLS.size()) > _RSLSreusedTimeWindows) {
696  int toRemove = _matrixCols_RSLS.back();
697  PRECICE_ASSERT(toRemove > 0, toRemove);
698  if (_matrixV_RSLS.size() > 0) {
699  PRECICE_ASSERT(_matrixV_RSLS.cols() > toRemove, _matrixV_RSLS.cols(), toRemove);
700  }
701 
702  // remove columns
703  for (int i = 0; i < toRemove; i++) {
704  utils::removeColumnFromMatrix(_matrixV_RSLS, _matrixV_RSLS.cols() - 1);
705  utils::removeColumnFromMatrix(_matrixW_RSLS, _matrixW_RSLS.cols() - 1);
706  }
707  _matrixCols_RSLS.pop_back();
708  }
709  _matrixCols_RSLS.push_front(0);
710  }
711 
712  //_info2<<'\n';
713 
714  // if efficient update of imvj is enabled
715  if (not _alwaysBuildJacobian || _imvjRestart) {
716  // need to apply the preconditioner, as all data structures are reverted after
717  // call to computeQNUpdate. Need to call this before the preconditioner is updated.
718 
719  // |= REBUILD QR-dec if needed ============|
720  // apply scaling to V, V' := P * V (only needed to reset the QR-dec of V)
721  _preconditioner->apply(_matrixV);
722 
723  if (_preconditioner->requireNewQR()) {
724  if (not(_filter == Acceleration::QR2FILTER)) { //for QR2 filter, there is no need to do this twice
725  _qrV.reset(_matrixV, getLSSystemRows());
726  }
727  _preconditioner->newQRfulfilled();
728  }
729  // apply the configured filter to the LS system
730  // as it changed in BaseQNAcceleration::iterationsConverged()
731  BaseQNAcceleration::applyFilter();
732  _preconditioner->revert(_matrixV);
733  // |=================== ============|
734 
735  // ------- RESTART/ JACOBIAN ASSEMBLY -------
736  if (_imvjRestart) {
737 
738  // add the matrices Wtil and Z of the converged configuration to the storage containers
739  Eigen::MatrixXd Z(_qrV.cols(), _qrV.rows());
740  // compute pseudo inverse using QR factorization and back-substitution
741  // also compensates for the scaling of V, i.e.,
742  // reverts Z' = R^-1 * Q^T * P^-1 as Z := Z' * P
743  pseudoInverse(Z);
744 
745  // push back unscaled pseudo Inverse, Wtil is also unscaled.
746  // all objects in Wtil chunk and Z chunk are NOT PRECONDITIONED
747  _WtilChunk.push_back(_Wtil);
748  _pseudoInverseChunk.push_back(Z);
749 
753  if (static_cast<int>(_WtilChunk.size()) >= _chunkSize + 1) {
754 
755  // < RESTART >
756  _nbRestarts++;
757  utils::Event restartUpdate("IMVJRestart");
758  restartIMVJ();
759  restartUpdate.stop();
760  }
761 
762  // only in imvj normal mode with efficient update:
763  } else {
764 
765  // compute explicit representation of Jacobian
766  buildJacobian();
767  }
768 
773  if (_timeWindowsReused > 0 || (_timeWindowsReused == 0 && _forceInitialRelaxation)) {
774  //_Wtil.conservativeResize(0, 0);
775  _resetLS = true;
776  }
777  }
778 
779  // only store Jacobian if imvj is in normal mode, i.e., the Jacobian is build
780  if (not _imvjRestart) {
781  // store inverse Jacobian from converged time window. NOT SCALED with preconditioner
782  _oldInvJacobian = _invJacobian;
783  }
784 }
