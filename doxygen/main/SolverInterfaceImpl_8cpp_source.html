394  double computedTimestepLength)
395 {
396 
397  PRECICE_TRACE(computedTimestepLength);
398 
399  // Events for the solver time, stopped when we enter, restarted when we leave advance
400  auto &solverEvent = EventRegistry::instance().getStoredEvent("solver.advance");
401  solverEvent.stop(precice::syncMode);
402  auto &solverInitEvent = EventRegistry::instance().getStoredEvent("solver.initialize");
403  solverInitEvent.stop(precice::syncMode);
404 
405  Event e("advance", precice::syncMode);
406  utils::ScopedEventPrefix sep("advance/");
407 
408  PRECICE_CHECK(_state != State::Constructed, "initialize() has to be called before advance().");
409  PRECICE_CHECK(_state != State::Finalized, "advance() cannot be called after finalize().")
410  PRECICE_ASSERT(_couplingScheme->isInitialized());
411  PRECICE_CHECK(isCouplingOngoing(), "advance() cannot be called when isCouplingOngoing() returns false.");
412  PRECICE_CHECK((not _couplingScheme->receivesInitializedData() && not _couplingScheme->sendsInitializedData()) || (_hasInitializedData),
413  "initializeData() needs to be called before advance if data has to be initialized.");
414  PRECICE_CHECK(!math::equals(computedTimestepLength, 0.0), "advance() cannot be called with a timestep size of 0.");
415  PRECICE_CHECK(computedTimestepLength > 0.0, "advance() cannot be called with a negative timestep size {}.", computedTimestepLength);
416  _numberAdvanceCalls++;
417 
418  // This is the first time advance is called. Initializes the waveform with data from initializeData or 0, if initializeData was not called.
419  // @todo: Can be moved to the end of initializeData(), if initializeData() becomes mandatory. See https://github.com/precice/precice/issues/1196.
420  if (_numberAdvanceCalls == 1) {
421  for (auto &context : _accessor->readDataContexts()) {
422  context.moveToNextWindow();
423  }
424  }
425 
426 #ifndef NDEBUG
427  PRECICE_DEBUG("Synchronize timestep length");
428  if (utils::IntraComm::isParallel()) {
429  syncTimestep(computedTimestepLength);
430  }
431 #endif
432 
433  double timeWindowSize = 0.0; // Length of (full) current time window
434  double timeWindowComputedPart = 0.0; // Length of computed part of (full) current time window
435  double time = 0.0; // Current time
436 
437  // Update the coupling scheme time state. Necessary to get correct remainder.
438  _couplingScheme->addComputedTime(computedTimestepLength);
439 
440  if (_couplingScheme->hasTimeWindowSize()) {
441  timeWindowSize = _couplingScheme->getTimeWindowSize();
442  timeWindowComputedPart = timeWindowSize - _couplingScheme->getThisTimeWindowRemainder();
443  } else {
444  // use time window size provided to advance, only allowed, if this participant sets the time window size for the other participant
445  timeWindowSize = computedTimestepLength;
446  timeWindowComputedPart = computedTimestepLength;
447  }
448 
449  time = _couplingScheme->getTime();
450 
451  if (_couplingScheme->willDataBeExchanged(0.0)) {
452  performDataActions({action::Action::WRITE_MAPPING_PRIOR}, time, computedTimestepLength, timeWindowComputedPart, timeWindowSize);
453  mapWrittenData();
454  performDataActions({action::Action::WRITE_MAPPING_POST}, time, computedTimestepLength, timeWindowComputedPart, timeWindowSize);
455  }
456 
457  PRECICE_DEBUG("Advance coupling scheme");
458  _couplingScheme->advance();
459 
460  if (_couplingScheme->isTimeWindowComplete()) {
461  for (auto &context : _accessor->readDataContexts()) {
462  context.moveToNextWindow();
463  }
464  }
465 
466  if (_couplingScheme->hasDataBeenReceived()) {
467  performDataActions({action::Action::READ_MAPPING_PRIOR}, time, computedTimestepLength, timeWindowComputedPart, timeWindowSize);
468  mapReadData();
469  performDataActions({action::Action::READ_MAPPING_POST}, time, computedTimestepLength, timeWindowComputedPart, timeWindowSize);
470  }
471 
472  if (_couplingScheme->isTimeWindowComplete()) {
473  performDataActions({action::Action::ON_TIME_WINDOW_COMPLETE_POST}, time, computedTimestepLength, timeWindowComputedPart, timeWindowSize);
474  }
475 
476  PRECICE_INFO(_couplingScheme->printCouplingState());
477 
478  PRECICE_DEBUG("Handle exports");
479  handleExports();
480 
481  resetWrittenData();
482 
483  _meshLock.lockAll();
484  solverEvent.start(precice::syncMode);
485  return _couplingScheme->getNextTimestepMaxLength();

900 void SolverInterfaceImpl::setMeshQuad(
901  MeshID meshID,
902  int firstEdgeID,
903  int secondEdgeID,
904  int thirdEdgeID,
905  int fourthEdgeID)
906 {
907  PRECICE_TRACE(meshID, firstEdgeID, secondEdgeID, thirdEdgeID,
908  fourthEdgeID);
909  PRECICE_CHECK(_dimensions == 3, "setMeshQuad is only possible for 3D cases. "
910  "Please set the dimension to 3 in the preCICE configuration file.");
911  PRECICE_REQUIRE_MESH_MODIFY(meshID);
912  MeshContext &context = _accessor->usedMeshContext(meshID);
913  if (context.meshRequirement == mapping::Mapping::MeshRequirement::FULL) {
914  mesh::PtrMesh &mesh = context.mesh;
915  using impl::errorInvalidEdgeID;
916  PRECICE_CHECK(mesh->isValidEdgeID(firstEdgeID), errorInvalidEdgeID(firstEdgeID));
917  PRECICE_CHECK(mesh->isValidEdgeID(secondEdgeID), errorInvalidEdgeID(secondEdgeID));
918  PRECICE_CHECK(mesh->isValidEdgeID(thirdEdgeID), errorInvalidEdgeID(thirdEdgeID));
919  PRECICE_CHECK(mesh->isValidEdgeID(fourthEdgeID), errorInvalidEdgeID(fourthEdgeID));
920 
921  PRECICE_CHECK(utils::unique_elements(utils::make_array(firstEdgeID, secondEdgeID, thirdEdgeID, fourthEdgeID)),
922  "The four edge ID's are not unique. Please check that the edges that form the quad are correct.");
923 
924  auto chain = mesh::asChain(utils::make_array(
925  &mesh->edges()[firstEdgeID], &mesh->edges()[secondEdgeID],
926  &mesh->edges()[thirdEdgeID], &mesh->edges()[fourthEdgeID]));
927  PRECICE_CHECK(chain.connected, "The four edges are not connect. Please check that the edges that form the quad are correct.");
928 
929  auto coords = mesh::coordsFor(chain.vertices);
930  PRECICE_CHECK(utils::unique_elements(coords),
931  "The four vertices that form the quad are not unique. "
932  "The resulting shape may be a point, line or triangle."
933  "Please check that the adapter sends the four unique vertices that form the quad, or that the mesh on the interface "
934  "is composed of planar quads.");
935 
936  auto convexity = math::geometry::isConvexQuad(coords);
937  PRECICE_CHECK(convexity.convex,
938  "The given quad is not convex. "
939  "Please check that the adapter send the four correct vertices or that the interface is composed of planar quads.");
940 
941  // Use the shortest diagonal to split the quad into 2 triangles.
942  // The diagonal to be used with edges (1, 2) and (0, 3) of the chain
943  double distance1 = (coords[0] - coords[2]).norm();
944  // The diagonal to be used with edges (0, 1) and (2, 3) of the chain
945  double distance2 = (coords[1] - coords[3]).norm();
946 
947  // The new edge, e[4], is the shortest diagonal of the quad
948  if (distance1 <= distance2) {
949  auto &diag = mesh->createUniqueEdge(*chain.vertices[0], *chain.vertices[2]);
950  mesh->createTriangle(*chain.edges[3], *chain.edges[0], diag);
951  mesh->createTriangle(*chain.edges[1], *chain.edges[2], diag);
952  } else {
953  auto &diag = mesh->createUniqueEdge(*chain.vertices[1], *chain.vertices[3]);
954  mesh->createTriangle(*chain.edges[0], *chain.edges[1], diag);
955  mesh->createTriangle(*chain.edges[2], *chain.edges[3], diag);
956  }
957  }
958 }

961  MeshID meshID,
962  int firstVertexID,
963  int secondVertexID,
964  int thirdVertexID,
965  int fourthVertexID)
966 {
967  PRECICE_TRACE(meshID, firstVertexID,
968  secondVertexID, thirdVertexID, fourthVertexID);
969  PRECICE_CHECK(_dimensions == 3, "setMeshQuadWithEdges is only possible for 3D cases."
970  " Please set the dimension to 3 in the preCICE configuration file.");
971  PRECICE_REQUIRE_MESH_MODIFY(meshID);
972  MeshContext &context = _accessor->usedMeshContext(meshID);

975  mesh::Mesh &mesh = *(context.mesh);
976  using impl::errorInvalidVertexID;
977  PRECICE_CHECK(mesh.isValidVertexID(firstVertexID), errorInvalidVertexID(firstVertexID));
978  PRECICE_CHECK(mesh.isValidVertexID(secondVertexID), errorInvalidVertexID(secondVertexID));
979  PRECICE_CHECK(mesh.isValidVertexID(thirdVertexID), errorInvalidVertexID(thirdVertexID));
980  PRECICE_CHECK(mesh.isValidVertexID(fourthVertexID), errorInvalidVertexID(fourthVertexID));
981 
982  auto vertexIDs = utils::make_array(firstVertexID, secondVertexID, thirdVertexID, fourthVertexID);
983  PRECICE_CHECK(utils::unique_elements(vertexIDs), "The four vertex ID's are not unique. Please check that the vertices that form the quad are correct.");
984 
985  auto coords = mesh::coordsFor(mesh, vertexIDs);
986  PRECICE_CHECK(utils::unique_elements(coords),

988  "Please check that the adapter sends the four unique vertices that form the quad, or that the mesh on the interface "
989  "is composed of quads. A mix of triangles and quads are not supported.");
990 
991  auto convexity = math::geometry::isConvexQuad(coords);
992  PRECICE_CHECK(convexity.convex, "The given quad is not convex. "
993  "Please check that the adapter send the four correct vertices or that the interface is composed of quads. "
994  "A mix of triangles and quads are not supported.");
995  auto reordered = utils::reorder_array(convexity.vertexOrder, mesh::vertexPtrsFor(mesh, vertexIDs));
996 
997  // Vertices are now in the order: V0-V1-V2-V3-V0.
998  // The order now identifies all outer edges of the quad.
999  auto &edge0 = mesh.createUniqueEdge(*reordered[0], *reordered[1]);
1000  auto &edge1 = mesh.createUniqueEdge(*reordered[1], *reordered[2]);
1001  auto &edge2 = mesh.createUniqueEdge(*reordered[2], *reordered[3]);
1002  auto &edge3 = mesh.createUniqueEdge(*reordered[3], *reordered[0]);
1003 
1004  // Use the shortest diagonal to split the quad into 2 triangles.
1005  // Vertices are now in V0-V1-V2-V3-V0 order. The new edge, e[4] is either 0-2 or 1-3
1006  double distance1 = (reordered[0]->getCoords() - reordered[2]->getCoords()).norm();
1007  double distance2 = (reordered[1]->getCoords() - reordered[3]->getCoords()).norm();
1008 
1009  // The new edge, e[4], is the shortest diagonal of the quad
1010  if (distance1 <= distance2) {
1011  auto &diag = mesh.createUniqueEdge(*reordered[0], *reordered[2]);
1012  mesh.createTriangle(edge0, edge1, diag);
1013  mesh.createTriangle(edge2, edge3, diag);
1014  } else {
1015  auto &diag = mesh.createUniqueEdge(*reordered[1], *reordered[3]);
1016  mesh.createTriangle(edge3, edge0, diag);
1017  mesh.createTriangle(edge1, edge2, diag);
1018  }
1019  }

1923 {

1925  //Originally, this was done in one loop. This however gave deadlock if two meshes needed to be communicated cross-wise.
1926  //Both loops need a different sorting
1927 
1928  auto &contexts = _accessor->usedMeshContexts();
1929 
1930  std::sort(contexts.begin(), contexts.end(),
1931  [](MeshContext const *const lhs, MeshContext const *const rhs) -> bool {

1933  });
1934 
1935  for (MeshContext *meshContext : contexts) {
1936  meshContext->partition->communicate();
1937  }
1938 
1939  // for two-level initialization, there is also still communication in partition::compute()
