18 bool lineIntersection(
19  const Eigen::Ref<const Eigen::Vector2d> &a,
20  const Eigen::Ref<const Eigen::Vector2d> &b,
21  const Eigen::Ref<const Eigen::Vector2d> &c,
22  const Eigen::Ref<const Eigen::Vector2d> &d,
23  Eigen::Ref<Eigen::Vector2d> & intersectionPoint)
24 {
25  // Compute denominator for solving 2x2 equation system
26  double D = a(0) * (d(1) - c(1)) +
27  b(0) * (c(1) - d(1)) +
28  d(0) * (b(1) - a(1)) -
29  c(0) * (a(1) - b(1));
30 
31  // If D==0, the two lines are parallel
32  if (math::equals(D, 0.0)) {
33  return false;
34  }
35 
36  // Compute line segment parameter s, which is in [0, 1], if the
37  // intersection point is within the segment.
38  double s = (a(0) * (d(1) - c(1)) + c(0) * (a(1) - d(1)) + d(0) * (c(1) - a(1))) / D;
39 
40  intersectionPoint = a + s * (b - a);
41  return true;
42 }

44 ResultConstants segmentPlaneIntersection(
45  const Eigen::Vector3d &pointOnPlane,
46  const Eigen::Vector3d &planeNormal,
47  const Eigen::Vector3d &firstPointSegment,
48  const Eigen::Vector3d &secondPointSegment,
49  Eigen::Vector3d & intersectionPoint)
50 {
51  // Methodology of "Computation Geometry in C", Joseph O'Rourke, Chapter 7.3.1
52 
53  // The plane is represented by: p(x,y,z) dot planeNormal = d (1)
54  // We need to compute d by using pointOnPlane as p(x,y,z):
55  double d = pointOnPlane.dot(planeNormal);
56 
57  // The segment is represented by:
58  // firstPointSegment + (secondPointSegment - firstPointSegment) * t (2)
59  // We insert (2) into (1) and reorder for t. The resulting equations reads
60  // t = (d - firstPointSegment dot planeNormal) /
61  // ((secondPointSegment - firstPointSegment) dot planeNormal). (3)
62  //
63  // If the denominator of (3) equals 0, the segment is parallel to the plane.
64  // If, in addition, the denominator of (3) equals 0, the segment lies in the
65  // plane. Otherwise we can compute a point of intersection.
66  Eigen::Vector3d segmentVec(secondPointSegment - firstPointSegment);
67  double nominator = d - firstPointSegment.dot(planeNormal);
68  double denominator = segmentVec.dot(planeNormal);
69  if (math::equals(denominator, 0.0)) {
70  if (math::equals(nominator, 0.0)) {
71  return CONTAINED;
72  } else {
73  return NO_INTERSECTION;
74  }
75  }
76  double t = nominator / denominator;
77 
78  // If t is larger than 1 or smaller than zero, the intersection is not within
79  // the line segment
80  if (math::greater(t, 1.0) || math::greater(0.0, t)) {
81  return NO_INTERSECTION;
82  }
83 
84  intersectionPoint = firstPointSegment + segmentVec * t;
85 
86  // If t equals 1 or 0, the segment is just touching the plane, otherwise, a
87  // real intersection is happening.
88  if (math::equals(t, 0.0) || math::equals(t, 1.0)) {
89  return TOUCHING;
90  }
91 
92  return INTERSECTION;
93 }

145 ConvexityResult isConvexQuad(std::array<Eigen::VectorXd, 4> coords)
146 {
147  /*
148  All points need to be projected into a new plane with only 2 coordinates, x' and y'. These are used to check
149  the convexity of the quad. These new coordinates are stored in 'coords'.
150  */
151  PRECICE_ASSERT(std::all_of(coords.cbegin(), coords.cend(),
152  [](const auto &v) { return v.size() == 3; }),
153  "This only works in 3D.");
154 
155  Eigen::Vector3d coordOrigin; // Origin point for the transformation of points onto the new plane
156  coordOrigin = coords[0];
157 
158  // Normal of the plane of first three points in the list of vertices
159  Eigen::Vector3d e_1 = coords[1] - coordOrigin;
160  Eigen::Vector3d e_2 = coords[2] - coordOrigin;
161  Eigen::Vector3d normalVector = e_1.cross(e_2);
162 
163  PRECICE_CHECK(math::equals(normalVector.dot(coords[3] - coordOrigin), 0.0),
164  "Non-planar quads are not supported. The vertex coordinates are: {}.", coords);
165 
166  //Transform Coordinates - coord[0] is the origin
167  for (int i = 0; i < 4; i++) {
168  Eigen::Vector3d coordinateDifference = coords[i] - coordOrigin;
169  coords[i][0] = e_1.dot(coordinateDifference);
170  coords[i][1] = e_2.dot(coordinateDifference);
171  coords[i][2] = normalVector.dot(coordinateDifference);
172  }
173 
174  /*
175  For the convex hull algorithm, the most left hand point regarding the x coordinate is chosen as the starting point.
176  The algorithm moves in an anti-clockwise position, finding the most right hand coordinate from the
177  previous most right hand point. The algorithm must find 3 other points in order to be a valid quad.
178  */
179 
180  //First find point with smallest x coord. This point must be in the convex set then and is the starting point of gift wrapping algorithm
181  int idLowestPoint = 0;
182  for (int i = 1; i < 4; i++) {
183  if (coords[i][0] < coords[idLowestPoint][0]) {
184  idLowestPoint = i;
185  }
186  }
187 
188  // Found starting point. Add this as the first vertex in the convex hull.
189  // current is the origin point => hull[0]
190  int validVertexIDCounter = 0; // Counts number of times a valid vertex is found. Must be 4 for a valid quad.
191  int currentVertex = idLowestPoint; // current valid vertex
192  ConvexityResult result{};
193  do {
194  // Add current point to result
195  result.vertexOrder[validVertexIDCounter] = currentVertex;
196 
197  // Next potential valid vertex
198  int nextVertex = (currentVertex + 1) % 4; // remainder resets loop through vector of points
199  for (int i = 0; i < 4; i++) {
200  double y1 = coords[currentVertex][1] - coords[nextVertex][1];
201  double y2 = coords[currentVertex][1] - coords[i][1];
202  double x1 = coords[currentVertex][0] - coords[nextVertex][0];
203  double x2 = coords[currentVertex][0] - coords[i][0];
204  double val = y2 * x1 - y1 * x2;
205 
206  if (val > 0) {
207  nextVertex = i;
208  }
209  }
210  // Now nextVertex is the most anti-clockwise with respect to current
211  // Set current as nextVertex for next iteration, so that nextVertex is added to
212  // result 'hull'
213  currentVertex = nextVertex;
214  validVertexIDCounter++;
215  } while (currentVertex != idLowestPoint); // While we don't come to first point
216 
217  //Ordering of quad is hull 0-1-2-3-0
218  result.convex = (validVertexIDCounter == 4);
219 
220  return result;
221 }
